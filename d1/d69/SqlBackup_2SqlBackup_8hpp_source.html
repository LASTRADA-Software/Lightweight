66/// The interface for progress updates.
67struct ProgressManager
68{
69 virtual ~ProgressManager() = default;

73 ProgressManager(ProgressManager&&) = default;
74 ProgressManager& operator=(ProgressManager&&) = default;
75
76 /// Gets called when the progress of an individual backup/restore operation changes.
