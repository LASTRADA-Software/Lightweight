20/// SQL fixed-capacity string that mimmicks standard library string/string_view with a fixed-size underlying
21/// buffer.
22///
23/// The underlying storage will not be guaranteed to be `\0`-terminated unless
24/// a call to mutable/const c_str() has been performed.
25///
26/// @ingroup DataTypes
27template <std::size_t N, typename T = char, SqlFixedStringMode Mode = SqlFixedStringMode::FIXED_SIZE>

229 template <std::size_t OtherSize, SqlFixedStringMode OtherMode>
230 LIGHTWEIGHT_FORCE_INLINE std::weak_ordering operator<=>(
231 SqlFixedString<OtherSize, T, OtherMode> const& other) const noexcept
232 {
233 if ((void*) this == (void*) &other) [[unlikely]]
234 return std::weak_ordering::equivalent;
235
236 for (std::size_t i = 0; i < (std::min)(size(), other.size()); ++i)
237 if (auto const cmp = _data[i] <=> other._data[i]; cmp != std::weak_ordering::equivalent) [[unlikely]]
238 return cmp;
239 return size() <=> other.size();
240 }

243 LIGHTWEIGHT_FORCE_INLINE constexpr bool operator==(
244 SqlFixedString<OtherSize, T, OtherMode> const& other) const noexcept
245 {
246 return (*this <=> other) == std::weak_ordering::equivalent;
247 }

328template <std::size_t N, typename T, SqlFixedStringMode Mode>
329struct SqlBasicStringOperations<SqlFixedString<N, T, Mode>>
330{
331 using CharType = T;
332 using ValueType = SqlFixedString<N, CharType, Mode>;
333 static constexpr auto ColumnType = []() constexpr -> SqlColumnTypeDefinition {
334 if constexpr (std::same_as<CharType, char>)
335 {
336 if constexpr (Mode == SqlFixedStringMode::VARIABLE_SIZE)
337 return SqlColumnTypeDefinitions::Varchar { N };
338 else
339 return SqlColumnTypeDefinitions::Char { N };
340 }
341 else
342 {
343 if constexpr (Mode == SqlFixedStringMode::VARIABLE_SIZE)
344 return SqlColumnTypeDefinitions::NVarchar { N };
345 else
346 return SqlColumnTypeDefinitions::NChar { N };
347 }
348 }();

371 {
372 str->reserve((std::min)(N, capacity));
373 str->resize((std::min)(N, capacity));
374 }
375
376 static void Resize(ValueType* str, SQLLEN indicator) noexcept

380
381 static void PostProcessOutputColumn(ValueType* result, SQLLEN indicator)
382 {
383 switch (indicator)
384 {
385 case SQL_NULL_DATA:
386 result->clear();
387 break;
388 case SQL_NO_TOTAL:
389 result->resize(N);
390 break;
391 default: {
392 auto const len = (std::min)(N, static_cast<std::size_t>(indicator) / sizeof(CharType));
393 result->setsize(len);
394
395 if constexpr (Mode == SqlFixedStringMode::FIXED_SIZE_RIGHT_TRIMMED)
396 {
397 TrimRight(result, indicator);
398 }
399 break;
400 }
401 }
402 }
403
404 LIGHTWEIGHT_FORCE_INLINE static void TrimRight(ValueType* boundOutputString, SQLLEN indicator) noexcept
405 {
406 size_t n = (std::min)((size_t) indicator, N - 1);
407 while (n > 0 && std::isspace((*boundOutputString)[n - 1]))
408 --n;
409 boundOutputString->setsize(n);
410 }
