16/// @brief A pre-rendered text block plus its line count, ready to be packed into
17/// one or more output files by `EmitChunked`.
18///
19/// The block is treated as opaque — `EmitChunked` never splits a block in two.
20/// Callers compute `lineCount` once and reuse the same value across multiple
21/// passes, since counting newlines on every visit would be wasteful for large
22/// migrations.
23struct CodeBlock
24{
25 std::string content;
26 std::size_t lineCount = 0;
27};
28
29/// @brief Greedy bin-packing of `blocks` across files of at most `maxLinesPerFile`
30/// lines.
31///
32/// One block always lands wholly in one chunk — even when its line count exceeds
33/// the budget — because builder DSL chains cannot be broken across translation
34/// units. The returned outer vector has no empty inner vectors. When
35/// `maxLinesPerFile == 0`, every block lands in a single chunk.
36[[nodiscard]] LIGHTWEIGHT_API std::vector<std::vector<CodeBlock>> GroupBlocksByLineBudget(
37 std::vector<CodeBlock> const& blocks, std::size_t maxLinesPerFile);
38
39/// @brief Result of an `EmitChunked` call.
40struct WriteResult
41{
42 /// All files actually written, in apply order. Either `[outputPath]` (single
43 /// file) or `[<stem>_part01.<ext>, <stem>_part02.<ext>, ...]`.
44 std::vector<std::filesystem::path> writtenFiles;
45};

47/// @brief Writes one or more output files, splitting `blocks` across `<stem>_partNN.<ext>`
48/// siblings when their combined line count exceeds `maxLinesPerFile`. When the
49/// total fits, a single `outputPath` is written and split is skipped.
50///
51/// `fileHeader` and `fileFooter` are emitted at the top/bottom of every produced
52/// file (so all chunks remain self-contained translation units when used for
53/// `.cpp` outputs). Pass empty strings to skip them.
54///
55/// @return `WriteResult` listing the paths actually written.
56/// @throws `std::runtime_error` if any output file cannot be opened.
57[[nodiscard]] LIGHTWEIGHT_API WriteResult EmitChunked(std::filesystem::path const& outputPath,
58 std::vector<CodeBlock> const& blocks,
59 std::size_t maxLinesPerFile,
60 std::string_view fileHeader = {},
61 std::string_view fileFooter = {});
