24/// @brief Represents an ODBC SQL error.
25///
26/// NOTE: This is a simple wrapper around the SQL return codes. It is not meant to be
27/// comprehensive, but rather to provide a simple way to convert SQL return codes to
28/// std::error_code.
29///
30/// The code below is DRAFT and may be subject to change.
31struct SqlErrorInfo
32{
33 SQLINTEGER nativeErrorCode {};
34 std::string sqlState = " "; // 5 characters + null terminator
35 std::string message;
36
37 /// Constructs an ODBC error info object from the given ODBC connection handle.
38 static SqlErrorInfo fromConnectionHandle(SQLHDBC hDbc)
39 {
40 return fromHandle(SQL_HANDLE_DBC, hDbc);
41 }
42
43 /// Constructs an ODBC error info object from the given ODBC statement handle.
44 static SqlErrorInfo fromStatementHandle(SQLHSTMT hStmt)
45 {
46 return fromHandle(SQL_HANDLE_STMT, hStmt);
47 }
48
49 /// Asserts that the given result is a success code, otherwise throws an exception.
50 static void RequireStatementSuccess(SQLRETURN result, SQLHSTMT hStmt, std::string_view message);
51
52 private:
53 static SqlErrorInfo fromHandle(SQLSMALLINT handleType, SQLHANDLE handle)
54 {
55 SqlErrorInfo info {};
56 info.message = std::string(1024, '\0');
57
58 SQLSMALLINT msgLen {};
59 SQLGetDiagRecA(handleType,
60 handle,
61 1,
62 (SQLCHAR*) info.sqlState.data(),
63 &info.nativeErrorCode,
64 (SQLCHAR*) info.message.data(),
65 (SQLSMALLINT) info.message.capacity(),
66 &msgLen);
67 info.message.resize(msgLen);
68 return info;
69 }
70};

103struct SqlErrorCategory: std::error_category
104{
105 static SqlErrorCategory const& get() noexcept
106 {
107 static SqlErrorCategory const category;
108 return category;
109 }
110 [[nodiscard]] char const* name() const noexcept override
111 {
112 return "Lightweight";
113 }
114
115 [[nodiscard]] std::string message(int code) const override
116 {
117 using namespace std::string_literals;
118 switch (static_cast<SqlError>(code))
119 {
120 case SqlError::SUCCESS:
121 return "SQL_SUCCESS"s;
122 case SqlError::SUCCESS_WITH_INFO:
123 return "SQL_SUCCESS_WITH_INFO"s;
124 case SqlError::NODATA:
125 return "SQL_NO_DATA"s;
126 case SqlError::FAILURE:
127 return "SQL_ERROR"s;
128 case SqlError::INVALID_HANDLE:
129 return "SQL_INVALID_HANDLE"s;
130 case SqlError::STILL_EXECUTING:
131 return "SQL_STILL_EXECUTING"s;
132 case SqlError::NEED_DATA:
133 return "SQL_NEED_DATA"s;
134 case SqlError::PARAM_DATA_AVAILABLE:
135 return "SQL_PARAM_DATA_AVAILABLE"s;
136 case SqlError::UNSUPPORTED_TYPE:
137 return "SQL_UNSUPPORTED_TYPE"s;
138 case SqlError::INVALID_ARGUMENT:
139 return "SQL_INVALID_ARGUMENT"s;
140 case SqlError::TRANSACTION_ERROR:
141 return "SQL_TRANSACTION_ERROR"s;
142 }
143 return std::format("SQL error code {}", code);
144 }
