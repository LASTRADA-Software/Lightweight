236 /// Creates a new table for the given record type.
237 template <typename Record>
238 SqlCreateTableQueryBuilder CreateTable()
239 {
240 static_assert(DataMapperRecord<Record>, "Record must satisfy DataMapperRecord");
241
242 auto builder = CreateTable(RecordTableName<Record>);
243
244#if defined(LIGHTWEIGHT_CXX26_REFLECTION)
245 constexpr auto ctx = std::meta::access_context::current();
246 template for (constexpr auto el: define_static_array(nonstatic_data_members_of(^^Record, ctx)))
247 {
248 using FieldType = typename[:std::meta::type_of(el):];
249 if constexpr (FieldWithStorage<FieldType>)
250 {
251 if constexpr (IsAutoIncrementPrimaryKey<FieldType>)
252 builder.PrimaryKeyWithAutoIncrement(std::string(FieldNameOf<el>),
253 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
254 else if constexpr (FieldType::IsPrimaryKey)
255 builder.PrimaryKey(std::string(FieldNameOf<el>),
256 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
257 else if constexpr (IsBelongsTo<FieldType>)
258 {
259 constexpr size_t referencedFieldIndex = []() constexpr -> size_t {
260 auto index = size_t(-1);
261 Reflection::EnumerateMembers<typename FieldType::ReferencedRecord>(
262 [&index]<size_t J, typename ReferencedFieldType>() constexpr -> void {
263 if constexpr (IsField<ReferencedFieldType>)
264 if constexpr (ReferencedFieldType::IsPrimaryKey)
265 index = J;
266 });
267 return index;
268 }();
269 builder.ForeignKey(
270 std::string(FieldNameOf<el>),
271 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>,
272 SqlForeignKeyReferenceDefinition {
273 .tableName = std::string { RecordTableName<typename FieldType::ReferencedRecord> },
274 .columnName = std::string { FieldNameOf<FieldType::ReferencedField> } });
275 }
276 else if constexpr (FieldType::IsMandatory)
277 builder.RequiredColumn(std::string(FieldNameOf<el>),
278 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
279 else
280 builder.Column(std::string(FieldNameOf<el>), SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
281 }
282 }
283#else
284 Reflection::EnumerateMembers<Record>([&builder]<size_t I, typename FieldType>() {
285 if constexpr (FieldWithStorage<FieldType>)
286 {
287 if constexpr (IsAutoIncrementPrimaryKey<FieldType>)
288 builder.PrimaryKeyWithAutoIncrement(std::string(FieldNameAt<I, Record>),
289 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
290 else if constexpr (FieldType::IsPrimaryKey)
291 builder.PrimaryKey(std::string(FieldNameAt<I, Record>),
292 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
293 else if constexpr (IsBelongsTo<FieldType>)
294 {
295 constexpr size_t referencedFieldIndex = []() constexpr -> size_t {
296 auto index = size_t(-1);
297 Reflection::EnumerateMembers<typename FieldType::ReferencedRecord>(
298 [&index]<size_t J, typename ReferencedFieldType>() constexpr -> void {
299 if constexpr (IsField<ReferencedFieldType>)
300 if constexpr (ReferencedFieldType::IsPrimaryKey)
301 index = J;
302 });
303 return index;
304 }();
305 builder.ForeignKey(
306 std::string(FieldNameAt<I, Record>),
307 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>,
308 SqlForeignKeyReferenceDefinition {
309 .tableName = std::string { RecordTableName<typename FieldType::ReferencedRecord> },
310 .columnName =
311 std::string { FieldNameAt<referencedFieldIndex, typename FieldType::ReferencedRecord> } });
312 }
313 else if constexpr (FieldType::IsMandatory)
314 builder.RequiredColumn(std::string(FieldNameAt<I, Record>),
315 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
316 else
317 builder.Column(std::string(FieldNameAt<I, Record>),
318 SqlColumnTypeDefinitionOf<typename FieldType::ValueType>);
319 }
320 });
321#endif
322 return builder;
323 }
