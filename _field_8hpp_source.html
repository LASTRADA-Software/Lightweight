51// Represents a single column in a table.
52//
53// The column name, index, and type are known at compile time.
54// If either name or index are not known at compile time, leave them at their default values,
55// but at least one of them msut be known.
56//
57// It is imperative that this data structure is an aggregate type, such that it works with C++20 reflection.
58template <detail::FieldElementType T, PrimaryKey IsPrimaryKeyValue = PrimaryKey::No>
59struct Field
60{
61 using ValueType = T;
62
63 // clang-format off
64 constexpr Field() noexcept = default;
65 constexpr Field(Field const&) noexcept = default;
66 constexpr Field& operator=(Field const&) noexcept = default;
67 constexpr Field(Field&&) noexcept = default;
68 constexpr Field& operator=(Field&&) noexcept = default;
69 constexpr ~Field() noexcept = default;
70 // clang-format on
71
72 template <typename... S>
73 requires std::constructible_from<T, S...>
74 constexpr Field(S&&... value) noexcept;
75
76 template <typename S>
77 requires std::constructible_from<T, S>
78 constexpr Field& operator=(S&& value) noexcept;
79
80 static constexpr auto IsOptional = detail::IsStdOptional<T>;
81 static constexpr auto IsMandatory = !IsOptional;
82 static constexpr auto IsPrimaryKey = IsPrimaryKeyValue != PrimaryKey::No;
83 static constexpr auto IsAutoIncrementPrimaryKey = IsPrimaryKeyValue == PrimaryKey::AutoIncrement;
84
85 constexpr std::weak_ordering operator<=>(Field const& other) const noexcept;
86
87 bool operator==(Field const& value) const noexcept = default;
88 bool operator!=(Field const& value) const noexcept = default;
89
90 bool operator==(T const& value) const noexcept;
91 bool operator!=(T const& value) const noexcept;
92
93 // Returns a string representation of the value, suitable for use in debugging and logging.
94 [[nodiscard]] std::string InspectValue() const;
95
96 void BindOutputColumn(SQLSMALLINT outputIndex, SqlStatement& stmt);
97
98 constexpr void SetModified(bool value) noexcept;
99 [[nodiscard]] constexpr bool IsModified() const noexcept;
100 [[nodiscard]] constexpr T const& Value() const noexcept;
101 [[nodiscard]] constexpr T& MutableValue() noexcept;
102
103 private:
104 ValueType _value {};
105 bool _modified { false };
106};

171template <detail::FieldElementType T, PrimaryKey IsPrimaryKey>
172inline LIGHTWEIGHT_FORCE_INLINE std::string Field<T, IsPrimaryKey>::InspectValue() const
173{
174 if constexpr (std::is_same_v<T, std::string>)
175 {
176 std::stringstream result;
177 result << std::quoted(_value, '\'');
178 return result.str();
179 }
180 else if constexpr (std::is_same_v<T, SqlText>)
181 {
182 std::stringstream result;
183 result << std::quoted(_value.value, '\'');
184 return result.str();
185 }
186 else if constexpr (std::is_same_v<T, SqlDate>)
187 return std::format("\'{}\'", _value.value);
188 else if constexpr (std::is_same_v<T, SqlTime>)
189 return std::format("\'{}\'", _value.value);
190 else if constexpr (std::is_same_v<T, SqlDateTime>)
191 return std::format("\'{}\'", _value.value());
192 else
193 return std::format("{}", _value);
194}

229template <detail::FieldElementType T, PrimaryKey IsPrimaryKey>
230struct SqlDataBinder<Field<T, IsPrimaryKey>>
231{
232 using ValueType = Field<T, IsPrimaryKey>;
233
234 static constexpr auto ColumnType = SqlDataBinder<T>::ColumnType;
235
236 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN InputParameter(SQLHSTMT stmt,
237 SQLUSMALLINT column,
238 ValueType const& value,
239 SqlDataBinderCallback& cb)
240 {
241 return SqlDataBinder<T>::InputParameter(stmt, column, value.Value(), cb);
242 }
243
244 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN
245 OutputColumn(SQLHSTMT stmt, SQLUSMALLINT column, ValueType* result, SQLLEN* indicator, SqlDataBinderCallback& cb)
246 {
247 auto const sqlReturn = SqlDataBinder<T>::OutputColumn(stmt, column, &result->MutableValue(), indicator, cb);
248 cb.PlanPostProcessOutputColumn([result]() { result->SetModified(true); });
249 return sqlReturn;
250 }
251
252 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN GetColumn(SQLHSTMT stmt,
253 SQLUSMALLINT column,
254 ValueType* result,
255 SQLLEN* indicator,
256 SqlDataBinderCallback const& cb) noexcept
257 {
258 auto const sqlReturn = SqlDataBinder<T>::GetColumn(stmt, column, &result->emplace(), indicator, cb);
259 if (SQL_SUCCEEDED(sqlReturn))
260 result->SetModified(true);
261 return sqlReturn;
262 }
263
264 static LIGHTWEIGHT_FORCE_INLINE std::string Inspect(ValueType const& value) noexcept
265 {
266 return value.InspectValue();
267 }
268};
