22/// Enum to define growth strategies of the pool
23///
24enum class GrowthStrategy : uint8_t
25{
26 /// Pre-create initialSize objects. Allow the total count
27 /// to grow up to maxSize. Once maxSize objects exist, callers BLOCK
28 /// until one is returned.
29 BoundedWait,
30
31 /// Pre-create initialSize objects. The pool stores up to maxSize
32 /// objects. If none are idle, a fresh object is ALWAYS created (no
33 /// waiting). On return: kept if the idle set is below maxSize, otherwise
34 /// destroyed.
35 BoundedOverflow,
36
37 /// Pre-create initialSize objects. Grow without limit. Every returned
38 /// object is always kept in the pool.
39 UnboundedGrow,
40};
41
42/// Structure to hold the configuration of the pool, including the initial size, maximum size and growth strategy.
43/// Structure is used as a template parameter for the Pool class to configure its behavior at compile time.
44struct PoolConfig
45{
46 /// Initial number of data mappers to pre-create and store in the pool, must be less than or equal to maxSize
47 size_t initialSize {};
48 /// Maximum number of data mappers that can exist in the pool, must be greater than or equal to initialSize
49 /// this is used for the Bounded* strategies to determine when to block or when to stop accepting returned data mappers,
50 /// for the UnboundedGrow strategy this is ignored
51 size_t maxSize {};
52 /// Strategy to determine how the pool should grow when there are no idle data mappers available, default is BoundedWait
53 /// which blocks until a data mapper is returned to the pool
54 GrowthStrategy growthStrategy { GrowthStrategy::BoundedWait };
55};

131 /// for bounded wait strategy, return the data mapper to the pool. A suspended async waiter (if any)
132 /// is handed the mapper directly and resumed; otherwise a blocked synchronous waiter is notified.
133 void Return(std::unique_ptr<DataMapper> dm) noexcept
134 requires(Config.growthStrategy == GrowthStrategy::BoundedWait)
135 {
136#if defined(LIGHTWEIGHT_ENABLE_ASYNC)
137 AsyncWaiter waiter;
138 bool haveAsyncWaiter = false;
139#endif
140 {

144 {
145 waiter = _asyncWaiters.front();
146 _asyncWaiters.pop_front();

148 haveAsyncWaiter = true;

167 /// for bounded overflow strategy, only return to pool if we have capacity, otherwise just destroy the data mapper
168 void Return(std::unique_ptr<DataMapper> dm) noexcept

196 /// Function to acquire a data mapper from the pool, the behavior of this function depends on the growth strategy
197 /// this is a specific implementation for the BoundedWait strategy, which blocks until a data mapper is available if the
198 /// pool is at maximum capacity
199 PooledDataMapper Acquire()
200 requires(Config.growthStrategy == GrowthStrategy::BoundedWait)
201 {
202 std::unique_lock lock(_mutex);
203 if (_idleDataMappers.empty())
204 {
205 if (_checkedOut >= Config.maxSize)
206 {
207 // wait until a data mapper is returned to the pool
208 _cv.wait(lock, [this] { return !_idleDataMappers.empty(); });
209 }
210 else
211 {
212 // create a new data mapper and return it
213 ++_checkedOut;
214 return PooledDataMapper(*this, std::make_unique<DataMapper>());
215 }
216 }
217
218 // get a data mapper from the pool
219 auto dm = std::move(_idleDataMappers.back());
220 _idleDataMappers.pop_back();
221 ++_checkedOut;
222 return PooledDataMapper(*this, std::move(dm));
223 }

225 /// Function to acquire a data mapper from the pool, the behavior of this function depends on the growth strategy
226 /// this is a specific implementation for the strategies that do not block, which always creates a new data mapper if
227 /// the pool is empty, regardless of the maximum capacity
228 PooledDataMapper Acquire()
229 requires(Config.growthStrategy != GrowthStrategy::BoundedWait)
230 {
231 std::scoped_lock lock(_mutex);
232 if (_idleDataMappers.empty())
233 {
234 // create a new data mapper and return it
235 return PooledDataMapper(*this, std::make_unique<DataMapper>());
236 }
237
238 // get a data mapper from the pool
239 auto dm = std::move(_idleDataMappers.back());
240 _idleDataMappers.pop_back();
241 return PooledDataMapper(*this, std::move(dm));
242 }

280 /// Awaitable that acquires a DataMapper, suspending only when the pool is at capacity.
281 struct AsyncAcquireAwaitable
282 {
283 Pool& pool;
284 Async::IResumeScheduler& resume;
285 std::unique_ptr<DataMapper> acquired {};
286
287 [[nodiscard]] static bool await_ready() noexcept
288 {
289 return false;
290 }
291
292 bool await_suspend(std::coroutine_handle<> handle)
293 {
294 std::scoped_lock const lock(pool._mutex);
295 if (!pool._idleDataMappers.empty())
296 {
297 acquired = std::move(pool._idleDataMappers.back());
298 pool._idleDataMappers.pop_back();
299 if constexpr (Config.growthStrategy == GrowthStrategy::BoundedWait)
300 ++pool._checkedOut;
301 return false; // do not suspend — resume immediately
302 }
303 if constexpr (Config.growthStrategy == GrowthStrategy::BoundedWait)
304 {
305 if (pool._checkedOut >= Config.maxSize)
306 {
307 pool._asyncWaiters.push_back(AsyncWaiter { handle, &resume, &acquired });
308 return true; // suspend until a mapper is returned
309 }
310 ++pool._checkedOut;
311 }
312 acquired = std::make_unique<DataMapper>();
313 return false;
314 }
315
316 std::unique_ptr<DataMapper> await_resume() noexcept
317 {
318 return std::move(acquired);
319 }
320 };
