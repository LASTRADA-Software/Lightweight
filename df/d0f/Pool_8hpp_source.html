20/// Enum to define growth strategies of the pool
21///
22enum class GrowthStrategy : uint8_t
23{
24 /// Pre-create initialSize objects. Allow the total count
25 /// to grow up to maxSize. Once maxSize objects exist, callers BLOCK
26 /// until one is returned.
27 BoundedWait,
28
29 /// Pre-create initialSize objects. The pool stores up to maxSize
30 /// objects. If none are idle, a fresh object is ALWAYS created (no
31 /// waiting). On return: kept if the idle set is below maxSize, otherwise
32 /// destroyed.
33 BoundedOverflow,
34
35 /// Pre-create initialSize objects. Grow without limit. Every returned
36 /// object is always kept in the pool.
37 UnboundedGrow,
38};

40/// Structure to hold the configuration of the pool, including the initial size, maximum size and growth strategy.
41/// Structure is used as a template parameter for the Pool class to configure its behavior at compile time.
42struct PoolConfig
43{
44 /// Initial number of data mappers to pre-create and store in the pool, must be less than or equal to maxSize
45 size_t initialSize {};
46 /// Maximum number of data mappers that can exist in the pool, must be greater than or equal to initialSize
47 /// this is used for the Bounded* strategies to determine when to block or when to stop accepting returned data mappers,
48 /// for the UnboundedGrow strategy this is ignored
49 size_t maxSize {};
50 /// Strategy to determine how the pool should grow when there are no idle data mappers available, default is BoundedWait
51 /// which blocks until a data mapper is returned to the pool
52 GrowthStrategy growthStrategy { GrowthStrategy::BoundedWait };
53};

134 /// for bounded wait strategy, return the data mapper to the pool. A suspended async waiter (if any)
135 /// is handed the mapper directly and resumed; otherwise a blocked synchronous waiter is notified.
136 void Return(std::unique_ptr<DataMapper> dm) noexcept
137 requires(Config.growthStrategy == GrowthStrategy::BoundedWait)
138 {
139 // Drop any async backend before the mapper is idled or handed to a waiter, so a recycled
140 // connection never carries references to (possibly destroyed) executors; a waiter's
141 // AcquireAsync re-enables it fresh.
142 dm->Connection().DisableAsync();
143 std::shared_ptr<AsyncWaiterNode> toResume;
144 {
145 std::scoped_lock const lock(_mutex);
146 toResume = ReturnLocked(std::move(dm));
147 }
148 // Resume outside the lock to avoid re-entrancy (the resumed coroutine may call back into the pool).
149 if (toResume)
150 toResume->resume->Resume(toResume->handle);
151 }

153 /// Hands @p dm to the next parked async waiter (transferring the checked-out count) or idles it.
154 ///
155 /// @pre @c _mutex is held by the caller.
156 /// @param dm The mapper to return; its async backend must already be disabled.
157 /// @return The waiter node that was handed the mapper and must be resumed by the caller @e after
158 /// releasing @c _mutex, or @c nullptr if the mapper was returned to the idle set.
159 std::shared_ptr<AsyncWaiterNode> ReturnLocked(std::unique_ptr<DataMapper> dm) noexcept
160 requires(Config.growthStrategy == GrowthStrategy::BoundedWait)
161 {
162 while (!_asyncWaiters.empty())
163 {
164 auto node = _asyncWaiters.front();
165 _asyncWaiters.pop_front();
166 // _asyncWaiters only ever holds parked nodes (an awaitable de-registers itself on
167 // abandonment), but guard defensively before handing off.
168 if (node->state == AsyncWaiterNode::State::Parked)
169 {
170 node->state = AsyncWaiterNode::State::Fulfilled;
171 node->mapper = std::move(dm); // hand off ownership; _checkedOut stays (transferred)
172 return node;
173 }
174 }
175 _idleDataMappers.push_back(std::move(dm));
176 --_checkedOut;
177 _cv.notify_one();
178 return nullptr;
179 }

181 /// for bounded overflow strategy, only return to pool if we have capacity, otherwise just destroy the data mapper

185 // Drop any async backend so a recycled connection never carries references to executors

203 /// Destructor. The pool manages the lifecycle of the idle data mappers; be aware that any
204 /// acquired data mappers not returned to the pool are destroyed when the pool is destroyed,
205 /// which may leak resources if not handled properly.
206 ~Pool() noexcept
207 {
208 // A coroutine still parked in AcquireAsync owns its own frame, which the pool can neither
209 // resume (it has no mapper to give it) nor destroy, and it holds a reference back to this
210 // pool — so destroying the pool out from under it is undefined. Drive every AcquireAsync task
211 // to completion (or destroy it) before destroying the pool. The shared AsyncWaiterNode design
212 // means a stale Return()/teardown never writes through freed bookkeeping, but a still-parked
213 // frame is intrinsically the caller's to finish; the assert flags violations in debug builds.
214 assert(_asyncWaiters.empty() && "Pool destroyed with coroutines still parked in AcquireAsync");
215 }

222 /// Function to acquire a data mapper from the pool, the behavior of this function depends on the growth strategy
223 /// this is a specific implementation for the BoundedWait strategy, which blocks until a data mapper is available if the
224 /// pool is at maximum capacity
225 PooledDataMapper Acquire()
226 requires(Config.growthStrategy == GrowthStrategy::BoundedWait)
227 {
228 std::unique_lock lock(_mutex);
229 if (_idleDataMappers.empty())
230 {
231 if (_checkedOut >= Config.maxSize)
232 {
233 // wait until a data mapper is returned to the pool
234 _cv.wait(lock, [this] { return !_idleDataMappers.empty(); });
235 }
236 else
237 {
238 // create a new data mapper and return it
239 ++_checkedOut;
240 return PooledDataMapper(*this, std::make_unique<DataMapper>());
241 }
242 }
243
244 // get a data mapper from the pool
245 auto dm = std::move(_idleDataMappers.back());
246 _idleDataMappers.pop_back();
247 ++_checkedOut;
248 return PooledDataMapper(*this, std::move(dm));
249 }

295 /// A suspended AcquireAsync coroutine waiting for a DataMapper to become available.
296 ///
297 /// Heap-allocated and co-owned (via @c std::shared_ptr) by the pool's @c _asyncWaiters queue and
298 /// the suspended @ref AsyncAcquireAwaitable. Keeping the handed-off mapper and the liveness
299 /// @c state in this node — rather than via a pointer into the coroutine frame — lets @ref Return
300 /// hand off and the awaitable's destructor coordinate purely through node state under @c _mutex,
301 /// never dereferencing a frame that may have been destroyed.
302 struct AsyncWaiterNode
303 {
304 /// Liveness of the waiter, transitioned only under @c Pool::_mutex.
305 enum class State : std::uint8_t
306 {
307 Parked, ///< Suspended and registered in @c _asyncWaiters, awaiting a mapper.
308 Fulfilled, ///< Return handed it a mapper (in @c mapper) and scheduled its resumption.
309 Abandoned, ///< The awaiting task was destroyed (or its mapper consumed); inert.
310 };
311
312 std::coroutine_handle<> handle {};
313 Async::IResumeScheduler* resume = nullptr;
314 std::unique_ptr<DataMapper> mapper {}; ///< Filled by Return on hand-off; lives outside the frame.
315 State state = State::Parked;
316 };

318 /// Awaitable that acquires a DataMapper, suspending only when the pool is at capacity.
319 ///
320 /// Non-copyable/non-movable: it is constructed in place in the co_await expression and lives in
321 /// the coroutine frame. When it has to suspend it registers a shared @ref AsyncWaiterNode in
322 /// pool._asyncWaiters; the node (not a pointer into this frame) carries the handed-off mapper and
323 /// the liveness state, so Return() and this awaitable's destructor coordinate safely under the
324 /// pool mutex even across threads.
325 struct AsyncAcquireAwaitable
326 {
327 Pool& pool;
328 Async::IResumeScheduler& resume;
329 std::unique_ptr<DataMapper> acquired {}; ///< Mapper obtained without suspending (idle/fresh).
330 std::shared_ptr<AsyncWaiterNode> node {}; ///< Set only while parked; shared with the pool.
331
332 AsyncAcquireAwaitable(Pool& poolRef, Async::IResumeScheduler& resumeRef) noexcept:
333 pool { poolRef },
334 resume { resumeRef }
335 {
336 }
337
338 AsyncAcquireAwaitable(AsyncAcquireAwaitable const&) = delete;
339 AsyncAcquireAwaitable& operator=(AsyncAcquireAwaitable const&) = delete;
340 AsyncAcquireAwaitable(AsyncAcquireAwaitable&&) = delete;
341 AsyncAcquireAwaitable& operator=(AsyncAcquireAwaitable&&) = delete;

343 /// Cleans up if the awaiting coroutine is destroyed before it consumes its mapper.
344 ///
345 /// Under pool._mutex: if still parked, de-registers the node so a later Return() never hands
346 /// off to a dead frame. If Return() already handed off a mapper (Fulfilled) that await_resume
347 /// never consumed, reclaims it into the pool so the BoundedWait checked-out count is not
348 /// leaked (possibly handing it straight to the next waiter, resumed after the lock is released).
349 ///
350 /// @warning A task that has already been handed a mapper must still be driven to completion:
351 /// the resumption Return() scheduled cannot be cancelled, so a coroutine frame with a pending
352 /// resumption must not be freed (do not destroy such a task concurrently with, or right after,
353 /// the hand-off). Likewise the pool must outlive every task acquired from it.
354 ~AsyncAcquireAwaitable()
355 {
356 if (!node)
357 return;
358 std::shared_ptr<AsyncWaiterNode> toResume;
359 {
360 std::scoped_lock const lock(pool._mutex);
361 switch (node->state)
362 {
363 case AsyncWaiterNode::State::Parked:
364 // Never fulfilled: remove ourselves so Return() won't hand off to a dead frame.
365 // Parking never incremented _checkedOut, so there is nothing to release.
366 node->state = AsyncWaiterNode::State::Abandoned;
367 std::erase(pool._asyncWaiters, node);
368 break;
369 case AsyncWaiterNode::State::Fulfilled:
370 // Handed a mapper but the task is dropped before consuming it: reclaim it,
371 // releasing this acquisition's checked-out count so the pool does not leak.
372 node->state = AsyncWaiterNode::State::Abandoned;
373 if constexpr (Config.growthStrategy == GrowthStrategy::BoundedWait)
374 {
375 if (node->mapper)
376 toResume = pool.ReturnLocked(std::move(node->mapper));
377 }
378 break;
379 case AsyncWaiterNode::State::Abandoned:
380 break;
381 }
382 }
383 if (toResume)
384 toResume->resume->Resume(toResume->handle);
385 }

392 bool await_suspend(std::coroutine_handle<> handle)
393 {
394 std::scoped_lock const lock(pool._mutex);
395 if (!pool._idleDataMappers.empty())
396 {
397 acquired = std::move(pool._idleDataMappers.back());
398 pool._idleDataMappers.pop_back();
399 if constexpr (Config.growthStrategy == GrowthStrategy::BoundedWait)
400 ++pool._checkedOut;
401 return false; // do not suspend — resume immediately
402 }
403 // Only BoundedWait bounds the pool and parks coroutines on exhaustion. The non-blocking
404 // strategies (BoundedOverflow — the default — and UnboundedGrow) always create a fresh
405 // mapper here, matching the synchronous Acquire() overloads, which also never suspend.
406 if constexpr (Config.growthStrategy == GrowthStrategy::BoundedWait)
407 {
408 if (pool._checkedOut >= Config.maxSize)
409 {
410 node = std::make_shared<AsyncWaiterNode>();
411 node->handle = handle;
412 node->resume = &resume;
413 pool._asyncWaiters.push_back(node);
414 return true; // suspend until a mapper is returned
415 }
416 ++pool._checkedOut;
417 }
418 acquired = std::make_unique<DataMapper>();
419 return false;
420 }
421
422 std::unique_ptr<DataMapper> await_resume() noexcept
423 {
424 // If we suspended, Return() placed the mapper in the shared node; take it here (on the
425 // resuming thread, with no concurrent access per the destruction contract). That leaves
426 // node->mapper empty, so the destructor treats the node as already consumed.
427 if (node)
428 return std::move(node->mapper);
429 return std::move(acquired);
430 }
431 };

433 Async::Task<PooledDataMapper> AcquireAsyncImpl(Async::IExecutor* dbWorkers, Async::IResumeScheduler* resume)
434 {
435 auto dm = co_await AsyncAcquireAwaitable { *this, *resume };
436 // Wrap in the RAII PooledDataMapper BEFORE the throwing EnableAsync call: if EnableAsync
437 // throws (e.g. bad_alloc), ~PooledDataMapper returns the mapper to the pool, decrementing
438 // _checkedOut and avoiding a permanent BoundedWait capacity leak.
439 auto pooled = PooledDataMapper(*this, std::move(dm));
440 pooled->Connection().EnableAsync(*dbWorkers, *resume);
441 co_return std::move(pooled);
442 }
