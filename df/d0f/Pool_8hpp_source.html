14/// Enum to define growth strategies of the pool
15///
16/// BoundedWait
17/// Pre-create initialSize objects. Allow the total count
18/// to grow up to maxSize. Once maxSize objects exist, callers BLOCK
19/// until one is returned.
20///
21/// BoundedOverflow
22/// Pre-create initialSize objects. The pool stores up to maxSize
23/// objects. If none are idle, a fresh object is ALWAYS created (no
24/// waiting). On return: kept if the idle set is below maxSize, otherwise
25/// destroyed.
26///
27/// UnboundedGrow
28/// Pre-create initialSize objects. Grow without limit. Every returned
29/// object is always kept in the pool.
30///
31enum class GrowthStrategy : uint8_t
32{
33 BoundedWait,
34 BoundedOverflow,
35 UnboundedGrow,
36};
37
38/// Structure to hold the configuration of the pool, including the initial size, maximum size and growth strategy.
39/// Structure is used as a template parameter for the Pool class to configure its behavior at compile time.
40struct PoolConfig
41{
42 size_t initialSize {};
43 size_t maxSize {};
44 GrowthStrategy growthStrategy { GrowthStrategy::BoundedWait };
45};

47/// A thread-save pool of DataMapper instances with the polisy configured by the PoolConfig template parameter.
48/// The pool allows acquiring and returning DataMapper instances, and manages the lifecycle of these instances according to
49/// the specified growth strategy.
50template <PoolConfig Config>
51class Pool
52{
53 public:
54 /// A wrapper around a DataMapper that returns it to the pool when destroyed
55 /// can be created only from the Pool and is move-only to ensure it is always
56 /// returned to the pool when it goes out of scope
57 class PooledDataMapper
58 {
59 friend class Pool;
60
61 explicit PooledDataMapper(Pool& pool, std::unique_ptr<DataMapper> dm) noexcept:
62 _dm { std::move(dm) },
63 _pool { pool }
64 {
65 }
66
67 public:
68 PooledDataMapper() = delete;
69 PooledDataMapper(PooledDataMapper const&) = delete;
70
71 /// Move constructor for the pooled data mapper, the only public
72 /// constructor, allows moving the pooled data mapper but not copying it
73 PooledDataMapper(PooledDataMapper&& other) noexcept:
74 _dm { std::move(other._dm) },
75 _pool { other._pool }
76 {
77 }
78 PooledDataMapper& operator=(PooledDataMapper const&) = delete;
79 PooledDataMapper& operator=(PooledDataMapper&&) = delete;
80 ~PooledDataMapper() noexcept
81 {
82 if (_dm)
83 ReturnToPool();
84 }
85
86 /// Access the underlying data mapper via pointer semantics
87 DataMapper* operator->() const noexcept
88 {
89 return _dm.get();
90 }
91
92 /// Access the underlying data mapper via reference semantics
93 /// This is useful for passing the pooled data mapper to functions
94 /// that expect a DataMapper reference
95 [[nodiscard]] DataMapper& get() const noexcept
96 {
97 return *_dm;
98 }
99
100 private:
101 void ReturnToPool() noexcept
102 {
103 _pool.Return(std::move(_dm));
104 _dm = nullptr;
105 }
106
107 std::unique_ptr<DataMapper> _dm;
108 Pool& _pool;
109 };
110
111 private:
112 /// always return the data mapper to the pool for this strategy
113 void Return(std::unique_ptr<DataMapper> dm) noexcept
114 requires(Config.growthStrategy == GrowthStrategy::UnboundedGrow)
115 {
116 std::scoped_lock lock(_mutex);
117 _idleDataMappers.push_back(std::move(dm));
118 }
119
120 /// for bounded wait strategy, return the data mapper to the pool and notify one waiting thread if any
121 void Return(std::unique_ptr<DataMapper> dm) noexcept
122 requires(Config.growthStrategy == GrowthStrategy::BoundedWait)
123 {
124 {
125 std::scoped_lock lock(_mutex);
126 _idleDataMappers.push_back(std::move(dm));
127 --_checkedOut;
128 }
129 _cv.notify_one();
130 }
131
132 /// for bounded overflow strategy, only return to pool if we have capacity, otherwise just destroy the data mapper
133 void Return(std::unique_ptr<DataMapper> dm) noexcept
134 requires(Config.growthStrategy == GrowthStrategy::BoundedOverflow)
135 {
136 std::scoped_lock lock(_mutex);
137 if (_idleDataMappers.size() < Config.maxSize)
138 _idleDataMappers.push_back(std::move(dm));
139 }
140
141 public:
142 /// Default constructor that pre-creates the initial number of data mappers and stores them in the pool
143 /// No other constructors are provided, as the pool is configured at compile time via the template parameter
144 explicit Pool()
145 {
146 _idleDataMappers.reserve(Config.initialSize);
147 for (size_t i = 0; i < Config.initialSize; ++i)
148 _idleDataMappers.push_back(std::make_unique<DataMapper>());
149 }
150
151 ~Pool() noexcept = default;
152
153 Pool(Pool const&) = delete;
154 Pool& operator=(Pool const&) = delete;
155 Pool(Pool&&) = delete;
156 Pool& operator=(Pool&&) = delete;
157
158 /// Function to acquire a data mapper from the pool, the behavior of this function depends on the growth strategy
159 /// this is a specific implementation for the BoundedWait strategy, which blocks until a data mapper is available if the
160 /// pool is at maximum capacity
161 PooledDataMapper Acquire()
162 requires(Config.growthStrategy == GrowthStrategy::BoundedWait)
163 {
164 std::unique_lock lock(_mutex);
165 if (_idleDataMappers.empty())
166 {
167 if (_checkedOut >= Config.maxSize)
168 {
169 // wait until a data mapper is returned to the pool
170 _cv.wait(lock, [this] { return !_idleDataMappers.empty(); });
171 }
172 else
173 {
174 // create a new data mapper and return it
175 ++_checkedOut;
176 return PooledDataMapper(*this, std::make_unique<DataMapper>());
177 }
178 }
179
180 // get a data mapper from the pool
181 auto dm = std::move(_idleDataMappers.back());
182 _idleDataMappers.pop_back();
183 ++_checkedOut;
184 return PooledDataMapper(*this, std::move(dm));
185 }
186
187 /// Function to acquire a data mapper from the pool, the behavior of this function depends on the growth strategy
188 /// this is a specific implementation for the strategeies that do not block, which always creates a new data mapper if
189 /// the pool is empty, regardless of the maximum capacity
190 PooledDataMapper Acquire()
191 requires(Config.growthStrategy != GrowthStrategy::BoundedWait)
192 {
193 std::scoped_lock lock(_mutex);
194 if (_idleDataMappers.empty())
195 {
196 // create a new data mapper and return it
197 return PooledDataMapper(*this, std::make_unique<DataMapper>());
198 }
199
200 // get a data mapper from the pool
201 auto dm = std::move(_idleDataMappers.back());
202 _idleDataMappers.pop_back();
203 return PooledDataMapper(*this, std::move(dm));
204 }
205
206#if defined(BUILD_TESTS)
207 [[nodiscard]] size_t IdleCount() noexcept
208 {
209 std::scoped_lock lock(_mutex);
210 return _idleDataMappers.size();
211 }
212#endif
213
214 private:
215 std::mutex _mutex;
216 std::condition_variable _cv;
217 std::vector<std::unique_ptr<DataMapper>> _idleDataMappers;
218 size_t _checkedOut {};
219};
