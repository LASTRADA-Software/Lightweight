3// `SecretResolver` is the front door for password / API-key lookups. Callers
4// (dbtool, the migrations GUI, downstream apps) hand it a `secretRef` string
5// like `env:ACME_PROD_PWD`, `file:~/.dbtool/acme.pwd` or a bare
6// `lightweight/acme-prod`; the resolver dispatches to the matching backend or
7// walks its backend chain in registration order.
8//
9// Platform-native backends (`keychain:` on macOS, `wincred:` on Windows,
10// `secretservice:` on Linux) live in separate translation units so that pure
11// CLI builds (e.g. `dbtool`) can ship with only `env:` / `file:` / `stdin:`
12// without pulling Qt or D-Bus in.
13//
14// The resolver is thread-compatible, not thread-safe: register backends at
15// start-up and only call `Resolve()` concurrently afterwards.

40/// Central lookup for opaque `secretRef` strings. Owns a list of registered
41/// backends and dispatches based on either an explicit `prefix:` or
42/// registration order for bare references.
43class LIGHTWEIGHT_API SecretResolver
44{
45 public:
46 SecretResolver() = default;
47 ~SecretResolver() = default;
48 SecretResolver(SecretResolver const&) = delete;
49 /// Move-construct from another resolver, taking ownership of its backend chain.
50 SecretResolver(SecretResolver&&) = default;
51 SecretResolver& operator=(SecretResolver const&) = delete;
52 /// Move-assign from another resolver, taking ownership of its backend chain.
53 SecretResolver& operator=(SecretResolver&&) = default;
54
55 /// Appends a backend to the chain. The backend's `Name()` is used both as
56 /// the prefix (`<name>:`) and for matching the fallback chain. Backends
57 /// registered later win when their prefix is explicitly requested, but
58 /// earlier registrations are tried first for bare references — giving
59 /// callers a clear way to shape the fallback order (register native OS
60 /// vaults first, file / env fallbacks last).
61 void RegisterBackend(std::shared_ptr<ISecretBackend> backend);
62
63 /// Looks up a secret by reference.
64 ///
65 /// `secretRef` is one of:
66 /// - `prefix:key` — route to the backend named `prefix`.
67 /// - `prefix:` — route to `prefix`, with `key` derived from
68 /// `profileName` (e.g. `stdin:` prompts interactively).
69 /// - bare key — walk the registered backend chain in order.
70 ///
71 /// `profileName` supplies a fallback key for backends that support it
72 /// (e.g. `env:` translates a bare ref into `LIGHTWEIGHT_<PROFILE>_PWD`).
73 ///
74 /// Returns `std::unexpected` when the prefix is unknown, the prefix is
75 /// registered but declines to build at runtime (e.g. `secretservice:` on
76 /// a headless box), or the bare-ref chain produced no hits.
77 [[nodiscard]] std::expected<std::string, ResolveError> Resolve(std::string_view secretRef,
78 std::string_view profileName) const;
79
80 /// Returns the registered backend prefixes in registration order, for
81 /// diagnostic / debugging output.
82 [[nodiscard]] std::vector<std::string> RegisteredBackendNames() const;
83
84 private:
85 [[nodiscard]] ISecretBackend* Lookup(std::string_view name) const noexcept;
86
87 [[nodiscard]] std::expected<std::string, ResolveError> ResolveExplicit(std::string_view prefix,
88 std::string_view key,
89 std::string_view profileName) const;
90
91 [[nodiscard]] std::expected<std::string, ResolveError> ResolveBare(std::string_view secretRef,
92 std::string_view profileName) const;
93
94 std::vector<std::shared_ptr<ISecretBackend>> _backends;
95};

97/// Convenience: builds a default resolver wired with the platform-neutral
98/// backends (`env:`, `file:`, `stdin:`) suitable for every build of
99/// Lightweight, regardless of Qt / libsecret availability.
100///
101/// Native GUI-only backends (`keychain:`, `wincred:`, `secretservice:`,
102/// `pass:`) are NOT registered here — consumers who need them must register
103/// them explicitly after the call. This keeps pure-CLI binaries like `dbtool`
104/// Qt-free.
105[[nodiscard]] LIGHTWEIGHT_API SecretResolver MakeDefaultResolver();
