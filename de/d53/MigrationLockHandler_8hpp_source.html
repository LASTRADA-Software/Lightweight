20/// Reason a migration-lock operation failed.
21///
22/// Carried inside `MigrationLockError` so the caller can decide whether to
23/// retry, escalate, or surface the cause to a human — without resorting to
24/// `what()`-string text matching.
25///
26/// @ingroup SqlMigration
27enum class MigrationLockFailureReason : uint8_t
28{
29 /// Lock wasn't acquired within the requested timeout.
30 Timeout,
31 /// Server picked us as the deadlock victim (e.g. SQL Server `sp_getapplock` = -3).
32 Deadlock,
33 /// Lock request cancelled by the server (e.g. SQL Server `sp_getapplock` = -2).
34 Cancelled,
35 /// Backend reported a parameter or validation error (e.g. SQL Server `sp_getapplock` = -999).
36 ParameterError,
37 /// Any other driver/server-level failure; `MigrationLockError::info` carries the SQLSTATE detail.
38 DriverError,
39};

41/// Structured error returned by `MigrationLockHandler::TryAcquire` and
42/// `MigrationLockHandler::Release`.
43///
44/// Lets callers propagate the *root cause* of a lock failure (timeout vs.
45/// deadlock vs. driver-level fault) instead of having to text-match an
46/// exception message.
47///
48/// @ingroup SqlMigration
49struct MigrationLockError
50{
51 /// What went wrong, at the granularity callers actually care about.
52 MigrationLockFailureReason reason {};
53 /// The lock name that was being acquired or released — handy when several
54 /// distinct locks coexist in a process.
55 std::string lockName;
56 /// Requested timeout (zero for `Release`).
57 std::chrono::milliseconds timeout {};
58 /// Pre-formatted, human-readable explanation. Already includes the lock
59 /// name and timeout where relevant, so callers can `std::println("{}", err.message)`
60 /// without further formatting.
61 std::string message;
62 /// Set on `DriverError`; carries SQLSTATE / native error / driver text.
63 std::optional<SqlErrorInfo> info;
64};

66/// Dialect-specific implementation of migration-lock acquire/release.
67///
68/// Each `SqlQueryFormatter` returns a process-singleton instance of the
69/// appropriate concrete handler via `SqlQueryFormatter::MigrationLockOps()`.
70/// Concrete implementations live next to the formatter that owns them
71/// (e.g. the SQL Server handler is defined alongside `SqlServerQueryFormatter`),
72/// keeping every dialect quirk in one source file and out of the
73/// `MigrationLock` business logic.
74///
75/// Callers that need a lock should use the `MigrationLock` RAII wrapper —
76/// this type is the extension point for adding new dialects.
77///
78/// @ingroup SqlMigration
79class [[nodiscard]] LIGHTWEIGHT_API MigrationLockHandler
80{
81 public:
82 MigrationLockHandler() = default;
83 /// Polymorphic destructor.
84 virtual ~MigrationLockHandler() = default;
85
86 MigrationLockHandler(MigrationLockHandler const&) = delete;
87 MigrationLockHandler& operator=(MigrationLockHandler const&) = delete;
88 MigrationLockHandler(MigrationLockHandler&&) = delete;
89 MigrationLockHandler& operator=(MigrationLockHandler&&) = delete;

91 /// Attempts to acquire the named lock on `connection`, blocking up to `timeout` ms.
92 ///
93 /// Returns an empty `expected` on success; otherwise a fully-populated
94 /// `MigrationLockError` so the caller can distinguish timeout from
95 /// deadlock from driver error and propagate accordingly.
96 ///
97 /// Implementations MUST NOT throw on a recoverable timeout — that's a
98 /// `MigrationLockFailureReason::Timeout`. They MAY throw only on truly
99 /// catastrophic conditions (out-of-memory, ABI mismatch); routine driver
100 /// failures should populate `MigrationLockFailureReason::DriverError`
101 /// with `info` set.
102 ///
103 /// @param connection The SQL connection to acquire the lock on.
104 /// @param lockName The name of the lock (advisory; backends hash or quote as needed).
105 /// @param timeout Maximum time to wait for lock acquisition.
106 /// @return Empty `expected` on success, populated `MigrationLockError` on failure.
107 [[nodiscard]] virtual std::expected<void, MigrationLockError> TryAcquire(SqlConnection& connection,
108 std::string_view lockName,
109 std::chrono::milliseconds timeout) const = 0;
110
111 /// Releases the named lock previously acquired with `TryAcquire`.
112 ///
113 /// Implementations MUST tolerate "release of a lock that's already gone"
114 /// (connection teardown, server-side session expiry, etc.) as success —
115 /// `Release` is idempotent by contract.
116 ///
117 /// Returns an error only if the release round-trip itself failed. The
118 /// caller (typically `MigrationLock`'s destructor) should log such an
119 /// error rather than swallow it silently.
120 ///
121 /// @param connection The SQL connection that holds the lock.
122 /// @param lockName The name of the lock to release.
123 /// @return Empty `expected` on success, populated `MigrationLockError` on failure.
124 [[nodiscard]] virtual std::expected<void, MigrationLockError> Release(SqlConnection& connection,
125 std::string_view lockName) const = 0;
126};
