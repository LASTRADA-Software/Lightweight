3// Multi-profile connection store, shared by dbtool and the migrations GUI.
4//
5// Extends the single-profile YAML format that `dbtool` used previously:
6// PluginsDir: ./migrations
7// ConnectionString: "DSN=acme-prod;UID=deploy"
8// Schema: dbo
9// backward-compatibly. Such a file loads as one anonymous profile called
10// "default". The new multi-profile format looks like:
11//
12// defaultProfile: acme-prod
13// profiles:
14// acme-prod:
15// pluginsDir: ./migrations
16// schema: dbo
17// dsn: ACME_PROD
18// uid: deploy
19// secretRef: lightweight/acme-prod
20// acme-dev:
21// connectionString: "Driver=SQLite3;Database=dev.db"
22//
23// Secret material (`password`) is never written to the YAML file. Callers
24// resolve `secretRef` via `Lightweight::Secrets::SecretResolver` before
25// attempting to open a connection.

43/// A single named connection profile.
44///
45/// Exactly one of `dsn` or `connectionString` is populated; the remaining
46/// field is left empty. `uid` and `secretRef` provide auth metadata that
47/// applies to both forms.
48struct Profile
49{
50 /// Profile name, unique within a ProfileStore.
51 std::string name;
52
53 /// Plugin search directory. Relative paths are resolved relative to
54 /// the current working directory at load time.
55 std::filesystem::path pluginsDir;
56
57 /// Optional database schema (e.g. "dbo").
58 std::string schema;
59
60 /// ODBC data source name. Mutually exclusive with `connectionString`.
61 std::string dsn;
62
63 /// Raw ODBC connection string. Mutually exclusive with `dsn`.
64 std::string connectionString;
65
66 /// Username used when connecting. Empty means "ask the driver manager".
67 std::string uid;
68
69 /// Opaque reference resolved by SecretResolver at connect time. May be
70 /// empty (no stored secret) or a prefixed ref like
71 /// `env:ACME_PROD_PWD`, `file:~/.dbtool/acme-prod.pwd`,
72 /// `keychain:lightweight/acme-prod`, `stdin:`.
73 std::string secretRef;
74
75 /// True when the profile carries enough info to attempt a connection.
76 [[nodiscard]] bool HasConnection() const noexcept
77 {
78 return !dsn.empty() || !connectionString.empty();
79 }
80
81 /// Builds an ODBC-level connect descriptor from this profile.
82 ///
83 /// `password` is the secret value previously resolved by the caller;
84 /// passing an empty string is valid (driver may prompt or use
85 /// integrated auth).
86 [[nodiscard]] LIGHTWEIGHT_API SqlConnectInfo ToConnectInfo(std::string_view password = {}) const;
87};

89/// In-memory collection of named `Profile`s plus "which is the default".
90class LIGHTWEIGHT_API ProfileStore
91{
92 public:
93 ProfileStore() = default;
94
95 /// Path where `LoadOrDefault()` looks when given no explicit path.
96 ///
97 /// - Windows: `%APPDATA%\dbtool\dbtool.yml`
98 /// - POSIX: `$XDG_CONFIG_HOME/dbtool/dbtool.yml` or
99 /// `$HOME/.config/dbtool/dbtool.yml`
100 [[nodiscard]] static std::filesystem::path DefaultPath();
101
102 /// Reads the store from `path`. Empty path means `DefaultPath()`.
103 /// A missing file yields an empty store (not an error); any other IO
104 /// or parse failure is reported via the error channel.
105 [[nodiscard]] static std::expected<ProfileStore, std::string> LoadOrDefault(std::filesystem::path path = {});
106
107 /// Writes the store back to `path`. Empty path means `DefaultPath()`.
108 /// Creates parent directories as needed. Never writes secret values.
109 [[nodiscard]] std::expected<void, std::string> Save(std::filesystem::path path = {}) const;
110
111 /// All profiles in insertion order.
112 [[nodiscard]] std::vector<Profile> const& Profiles() const noexcept
113 {
114 return _profiles;
115 }
116
117 /// Looks up a profile by name. Returns nullptr if none matches.
118 [[nodiscard]] Profile const* Find(std::string_view name) const noexcept;
119
120 /// Returns the default profile (by `DefaultProfileName()`), or the
121 /// first profile if no default is set, or nullptr if empty.
122 [[nodiscard]] Profile const* Default() const noexcept;
123
124 /// Name of the default profile. Empty when unset.
125 [[nodiscard]] std::string const& DefaultProfileName() const noexcept
126 {
127 return _defaultProfile;
128 }
129
130 /// Inserts or replaces a profile with the given name.
131 void Upsert(Profile profile);
132
133 /// Removes a profile by name. Returns true if removed.
134 bool Remove(std::string_view name);
135
136 /// Sets the default profile name. Pass "" to clear.
137 void SetDefault(std::string name)
138 {
139 _defaultProfile = std::move(name);
140 }
141
142 /// Number of profiles.
143 [[nodiscard]] std::size_t Size() const noexcept
144 {
145 return _profiles.size();
146 }
147
148 /// True when there are no profiles.
149 [[nodiscard]] bool Empty() const noexcept
150 {
151 return _profiles.empty();
152 }
153
154 private:
155 std::vector<Profile> _profiles;
156 std::string _defaultProfile;
157};
