17 /// One-shot completion signal used to bridge a coroutine back to a blocking caller.
18 ///
19 /// Supports both a blocking wait (@ref Wait, used by @ref SyncWait) and a pollable
20 /// flag (@ref IsSet, used by @ref SyncWaitPumping which drives an executor instead of
21 /// blocking).
22 class SyncWaitEvent
23 {
24 public:
25 void Set() noexcept
26 {
27 _set.store(true, std::memory_order_release);
28 _semaphore.release();
29 }
30
31 void Wait() noexcept
32 {
33 _semaphore.acquire();
34 }
35
36 [[nodiscard]] bool IsSet() const noexcept
37 {
38 return _set.load(std::memory_order_acquire);
39 }
40
41 private:
42 std::atomic<bool> _set { false };
43 std::binary_semaphore _semaphore { 0 };
44 };

49 /// Promise for the internal driver coroutine used by SyncWait.
50 ///
51 /// The completion signal is raised from the @c final_suspend awaiter (i.e. only once
52 /// the driver has fully suspended), so the blocking caller can safely destroy the
53 /// coroutine frame without racing the resuming thread.
54 template <typename T>
55 class SyncWaitPromise
56 {
57 public:
58 SyncWaitTask<T> get_return_object() noexcept;
59
60 static std::suspend_always initial_suspend() noexcept
61 {
62 return {};
63 }
64
65 static auto final_suspend() noexcept
66 {
67 struct FinalAwaiter
68 {
69 [[nodiscard]] static bool await_ready() noexcept
70 {
71 return false;
72 }
73 static void await_suspend(std::coroutine_handle<SyncWaitPromise> coro) noexcept
74 {
75 coro.promise().Signal();
76 }
77 static void await_resume() noexcept {}
78 };
79 return FinalAwaiter {};
80 }
81
82 void unhandled_exception() noexcept
83 {
84 _result.template emplace<2>(std::current_exception());
85 }
86
87 void return_value(T&& value) noexcept(std::is_nothrow_move_constructible_v<T>)
88 {
89 _result.template emplace<1>(std::move(value));
90 }
91
92 void return_value(T const& value)
93 {
94 _result.template emplace<1>(value);
95 }
96
97 [[nodiscard]] T Take()
98 {
99 if (_result.index() == 2)
100 std::rethrow_exception(std::get<2>(_result));
101 return std::move(std::get<1>(_result));
102 }
103
104 void Bind(SyncWaitEvent& signal) noexcept
105 {
106 _event = &signal;
107 }
108
109 void Signal() const noexcept
110 {
111 _event->Set();
112 }
113
114 private:
115 SyncWaitEvent* _event = nullptr;
116 std::variant<std::monostate, T, std::exception_ptr> _result;
117 };

176 template <typename T>
177 class SyncWaitTask
178 {
179 public:
180 using promise_type = SyncWaitPromise<T>;
181 using Handle = std::coroutine_handle<promise_type>;
182
183 explicit SyncWaitTask(Handle handle) noexcept:
184 _handle { handle }
185 {
186 }
187
188 SyncWaitTask(SyncWaitTask&& other) noexcept:
189 _handle { std::exchange(other._handle, {}) }
190 {
191 }
192
193 SyncWaitTask(SyncWaitTask const&) = delete;
194 SyncWaitTask& operator=(SyncWaitTask const&) = delete;
195 SyncWaitTask& operator=(SyncWaitTask&&) = delete;
196
197 ~SyncWaitTask()
198 {
199 if (_handle)
200 _handle.destroy();
201 }

204 {
205 _handle.promise().Bind(signal);

240/// Blocks the calling thread until @p task completes and returns its result (or rethrows).
241///
242/// Use this from non-coroutine code (e.g. @c main, tests) when the task resumes on a
243/// thread-backed scheduler. If the task resumes on a @ref ManualExecutor that only the
244/// calling thread pumps, use @ref SyncWaitPumping instead to avoid a deadlock.
245///
246/// @tparam T The task's result type.
247/// @param task The task to drive to completion (consumed).
248/// @return The task's result, or @c void.
249template <typename T>
250T SyncWait(Task<T> task)
251{
252 detail::SyncWaitEvent signal;
253 auto driver = detail::MakeSyncWaitTask<T>(std::move(task));
254 driver.Start(signal);
255 signal.Wait();
256 return driver.Promise().Take();
257}

259/// Drives @p task to completion by pumping @p executor on the calling thread.
260///
261/// This is the single-threaded counterpart to @ref SyncWait: rather than blocking, it runs
262/// the executor's queued work (including the task's resumption) until the task finishes.
263/// Intended for an app/event-loop thread and for deterministic tests using a
264/// @ref ManualExecutor.
265///
266/// @tparam T The task's result type.
267/// @tparam Executor An executor exposing @c RunUntil(predicate).
268/// @param task The task to drive to completion (consumed).
269/// @param executor The executor to pump (typically a @ref ManualExecutor).
270/// @return The task's result, or @c void.
271template <typename T, typename Executor>
272T SyncWaitPumping(Task<T> task, Executor& executor)
273{
274 detail::SyncWaitEvent signal;
275 auto driver = detail::MakeSyncWaitTask<T>(std::move(task));
276 driver.Start(signal);
277 executor.RunUntil([&signal] { return signal.IsSet(); });
278 return driver.Promise().Take();
279}
