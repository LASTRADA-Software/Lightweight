11class PostgreSqlFormatter final: public SQLiteQueryFormatter
12{
13 public:
14 [[nodiscard]] std::string QueryLastInsertId(std::string_view /*tableName*/) const override
15 {
16 // NB: Find a better way to do this on the given table.
17 // In our case it works, because we're expected to call this right after an insert.
18 // But a race condition may still happen if another client inserts a row at the same time too.
19 return std::format("SELECT lastval();");
20 }
21
22 [[nodiscard]] std::string BuildColumnDefinition(SqlColumnDeclaration const& column) const override
23 {
24 std::stringstream sqlQueryString;
25
26 sqlQueryString << '"' << column.name << "\" ";
27
28 if (column.primaryKey == SqlPrimaryKeyType::AUTO_INCREMENT)
29 sqlQueryString << "SERIAL";
30 else
31 sqlQueryString << ColumnType(column.type);
32
33 if (column.required)
34 sqlQueryString << " NOT NULL";
35
36 if (column.primaryKey == SqlPrimaryKeyType::AUTO_INCREMENT)
37 sqlQueryString << " PRIMARY KEY";
38
39 if (column.unique && !column.index)
40 sqlQueryString << " UNIQUE";
41
42 return sqlQueryString.str();
43 }
44
45 [[nodiscard]] std::string ColumnType(SqlColumnTypeDefinition const& type) const override
46 {
47 using namespace SqlColumnTypeDefinitions;
48
49 // PostgreSQL stores all strings as UTF-8
50 return std::visit(detail::overloaded {
51 [](Bigint const&) -> std::string { return "BIGINT"; },
52 [](Binary const& type) -> std::string { return std::format("BYTEA", type.size); },
53 [](Bool const&) -> std::string { return "BOOLEAN"; },
54 [](Char const& type) -> std::string { return std::format("CHAR({})", type.size); },
55 [](Date const&) -> std::string { return "DATE"; },
56 [](DateTime const&) -> std::string { return "TIMESTAMP"; },
57 [](Decimal const& type) -> std::string {
58 return std::format("DECIMAL({}, {})", type.precision, type.scale);
59 },
60 [](Guid const&) -> std::string { return "UUID"; },
61 [](Integer const&) -> std::string { return "INTEGER"; },
62 [](NChar const& type) -> std::string { return std::format("CHAR({})", type.size); },
63 [](NVarchar const& type) -> std::string { return std::format("VARCHAR({})", type.size); },
64 [](Real const&) -> std::string { return "REAL"; },
65 [](Smallint const&) -> std::string { return "SMALLINT"; },
66 [](Text const&) -> std::string { return "TEXT"; },
67 [](Time const&) -> std::string { return "TIME"; },
68 [](Timestamp const&) -> std::string { return "TIMESTAMP"; },
69 // NB: PostgreSQL doesn't have a TINYINT type, but it does have a SMALLINT type.
70 [](Tinyint const&) -> std::string { return "SMALLINT"; },
71 [](VarBinary const& type) -> std::string { return std::format("BYTEA({})", type.size); },
72 [](Varchar const& type) -> std::string { return std::format("VARCHAR({})", type.size); },
73 },
74 type);
75 }
76
77 [[nodiscard]] StringList AlterTable(std::string_view tableName,
78 std::vector<SqlAlterTableCommand> const& commands) const override
79 {
80 std::stringstream sqlQueryString;
81
82 int currentCommand = 0;
83 for (SqlAlterTableCommand const& command: commands)
84 {
85 if (currentCommand > 0)
86 sqlQueryString << '\n';
87 ++currentCommand;
88
89 using namespace SqlAlterTableCommands;
90 sqlQueryString << std::visit(
91 detail::overloaded {
92 [tableName](RenameTable const& actualCommand) -> std::string {
93 return std::format(
94 R"(ALTER TABLE "{}" RENAME TO "{}";)", tableName, actualCommand.newTableName);
95 },
96 [tableName, this](AddColumn const& actualCommand) -> std::string {
97 return std::format(R"(ALTER TABLE "{}" ADD COLUMN "{}" {} {};)",
98 tableName,
99 actualCommand.columnName,
100 ColumnType(actualCommand.columnType),
101 actualCommand.nullable == SqlNullable::NotNull ? "NOT NULL" : "NULL");
102 },
103 [tableName, this](AlterColumn const& actualCommand) -> std::string {
104 return std::format(
105 R"(ALTER TABLE "{0}" ALTER COLUMN "{1}" TYPE {2}, ALTER COLUMN "{1}" {3} NOT NULL;)",
106 tableName,
107 actualCommand.columnName,
108 ColumnType(actualCommand.columnType),
109 actualCommand.nullable == SqlNullable::NotNull ? "SET" : "DROP");
110 },
111 [tableName](RenameColumn const& actualCommand) -> std::string {
112 return std::format(R"(ALTER TABLE "{}" RENAME COLUMN "{}" TO "{}";)",
113 tableName,
114 actualCommand.oldColumnName,
115 actualCommand.newColumnName);
116 },
117 [tableName](DropColumn const& actualCommand) -> std::string {
118 return std::format(
119 R"(ALTER TABLE "{}" DROP COLUMN "{}";)", tableName, actualCommand.columnName);
120 },
121 [tableName](AddIndex const& actualCommand) -> std::string {
122 using namespace std::string_view_literals;
123 auto const uniqueStr = actualCommand.unique ? "UNIQUE "sv : ""sv;
124 return std::format(R"(CREATE {2}INDEX "{0}_{1}_index" ON "{0}"("{1}");)",
125 tableName,
126 actualCommand.columnName,
127 uniqueStr);
128 },
129 [tableName](DropIndex const& actualCommand) -> std::string {
130 return std::format(R"(DROP INDEX "{0}_{1}_index";)", tableName, actualCommand.columnName);
131 },
132 [tableName](AddForeignKey const& actualCommand) -> std::string {
133 return std::format(
134 R"(ALTER TABLE "{}" ADD {};)",
135 tableName,
136 BuildForeignKeyConstraint(actualCommand.columnName, actualCommand.referencedColumn));
137 },
138 [tableName](DropForeignKey const& actualCommand) -> std::string {
139 return std::format(R"(ALTER TABLE "{}" DROP CONSTRAINT "{}";)",
140 tableName,
141 std::format("FK_{}", actualCommand.columnName));
142 },
143 },
144 command);
145 }
146
147 return { sqlQueryString.str() };
148 }
149};
