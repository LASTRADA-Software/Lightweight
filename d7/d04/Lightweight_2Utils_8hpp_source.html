167 template <auto ReferencedField>
168 struct QuotedFieldNameOfImpl
169 {
170 static constexpr auto ClassName = RecordTableName<MemberClassType<decltype(ReferencedField)>>;
171 static constexpr auto FieldName = FieldNameOf<ReferencedField>;
172 static constexpr auto StorageSize = ClassName.size() + FieldName.size() + 6;
173
174 // Holds the full field name in the format "ClassName"."FieldName"
175 static constexpr auto Storage = []() constexpr -> std::array<char, StorageSize> {
176 // clang-format off
177 auto storage = std::array<char, StorageSize> {};
178 std::ranges::copy("\"", storage.begin());
179 std::ranges::copy(ClassName, storage.begin() + 1);
180 std::ranges::copy("\".\"", storage.begin() + 1 + ClassName.size());
181 std::ranges::copy(FieldName, storage.begin() + 1 + ClassName.size() + 3);
182 std::ranges::copy("\"", storage.begin() + 1 + ClassName.size() + 3 + FieldName.size());
183 storage.back() = '\0';
184 // clang-format on
185 return storage;
186 }();
187 static constexpr auto value = std::string_view(Storage.data(), Storage.size() - 1);
188 };

259 template <auto... ReferencedFields>
260 struct QuotedFieldNamesOfImpl
261 {
262 static constexpr auto StorageSize =
263 1 + (2 * (sizeof...(ReferencedFields) - 1)) + (0 + ... + QuotedFieldNameOf<ReferencedFields>.size());
264
265 static constexpr std::array<char, StorageSize> Storage = []() consteval {
266 auto result = std::array<char, StorageSize> {};
267 size_t offset = 0;
268 (
269 [&] {
270 if (offset > 0)
271 {
272 constexpr auto Delimiter = std::string_view(", ");
273 std::ranges::copy(Delimiter, result.begin() + offset);
274 offset += Delimiter.size();
275 }
276 std::ranges::copy(QuotedFieldNameOf<ReferencedFields>, result.begin() + offset);
277 offset += QuotedFieldNameOf<ReferencedFields>.size();
278 }(),
279 ...);
280 result.back() = '\0';
281 return result;
282 }();
283
284 static constexpr auto value = std::string_view(Storage.data(), Storage.size() - 1);
285 };
