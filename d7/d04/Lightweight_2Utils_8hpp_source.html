140template <std::size_t I, typename Record>
141consteval std::string_view FieldNameAt()
142{
143 using FieldType = Reflection::MemberTypeOf<I, Record>;
144
145 if constexpr (!std::string_view(ColumnNameOverride<FieldType>).empty())
146 {
147 return FieldType::ColumnNameOverride;
148 }
149 else if constexpr (requires { FieldType::ReferencedField; }) // check isBelongsTo
150 {
151 return detail::BelongsToNameImpl<I, Record>::name;
152 }
153 else
154 return Reflection::MemberNameOf<I, Record>;
155}

185template <auto ReferencedField>
186struct FullFieldNameOfImpl
187{
188 static constexpr auto ClassName = RecordTableName<MemberClassType<decltype(ReferencedField)>>;
189 static constexpr auto FieldName = FieldNameOf<ReferencedField>;
190 static constexpr auto StorageSize = ClassName.size() + FieldName.size() + 6;
191
192 // Holds the full field name in the format "ClassName"."FieldName"
193 static constexpr auto Storage = []() constexpr -> std::array<char, StorageSize> {
194 // clang-format off
195 auto storage = std::array<char, StorageSize> {};
196 std::ranges::copy("\"", storage.begin());
197 std::ranges::copy(ClassName, storage.begin() + 1);
198 std::ranges::copy("\".\"", storage.begin() + 1 + ClassName.size());
199 std::ranges::copy(FieldName, storage.begin() + 1 + ClassName.size() + 3);
200 std::ranges::copy("\"", storage.begin() + 1 + ClassName.size() + 3 + FieldName.size());
201 storage.back() = '\0';
202 // clang-format on
203 return storage;
204 }();
205 static constexpr auto value = std::string_view(Storage.data(), Storage.size() - 1);
206};
207} // namespace detail
