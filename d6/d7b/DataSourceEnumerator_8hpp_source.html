3// Thin wrapper over the ODBC driver manager's `SQLDataSources` / `SQLDrivers`
4// entry points. Used by the migrations GUI (to populate the DSN picker) and
5// exposed as a library-level utility so other Lightweight-based apps can reuse
6// it without copy-pasting the handle lifetime plumbing.
7//
8// The enumerator does not require an open `SqlConnection`: it operates on its
9// own scratch `SQL_HANDLE_ENV`. Queries are synchronous and can be slow on
10// Windows when there are many installed drivers — call from a worker thread
11// in GUI contexts.
12

25/// A single ODBC data source as reported by `SQLDataSources`.
26struct DataSourceInfo
27{
28 /// DSN name, suitable for use in `DRIVER=...;DSN=<name>;…`.
29 std::string name;
30
31 /// Driver description string (the driver's friendly name, e.g. "SQL Server"
32 /// or "ODBC Driver 18 for SQL Server"). May be empty if the driver manager
33 /// declined to report one.
34 std::string description;
35
36 /// Scope of the data source registration — user-private or system-wide.
37 enum class Scope : std::uint8_t
38 {
39 User,
40 System,
41 };
42 Scope scope = Scope::User;
43
44 auto operator<=>(DataSourceInfo const&) const noexcept = default;
45};
46
47/// A single installed ODBC driver as reported by `SQLDrivers`.
48struct DriverInfo
49{
50 /// Driver name, e.g. "ODBC Driver 18 for SQL Server".
51 std::string name;
52
53 /// Driver attribute key/value pairs as returned by the driver manager.
54 /// Typical keys include "FileUsage", "APILevel", "ConnectFunctions".
55 std::vector<std::pair<std::string, std::string>> attributes;
56
57 auto operator<=>(DriverInfo const&) const noexcept = default;
58};
59
60/// Enumerates every ODBC data source visible to the driver manager.
61///
62/// On success the returned vector contains user DSNs first, then system DSNs,
63/// in the order the driver manager reports them. On failure (driver manager
64/// not installed, environment handle allocation failed) the vector is empty.
65///
66/// The call is synchronous. Call from a worker thread if you need to keep a
67/// GUI event loop responsive.
68[[nodiscard]] LIGHTWEIGHT_API std::vector<DataSourceInfo> EnumerateDataSources();
69
70/// Enumerates every ODBC driver installed on the system.
71///
72/// Same semantics as `EnumerateDataSources` — synchronous, empty vector on
73/// failure.
74[[nodiscard]] LIGHTWEIGHT_API std::vector<DriverInfo> EnumerateDrivers();
75
76} // namespace Lightweight::Odbc
