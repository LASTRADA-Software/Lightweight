17/// @brief Tells the data mapper that this field is a primary key with given semantics, or not a primary key.
18enum class PrimaryKey : uint8_t
19{
20 /// The field is not a primary key.
21 No,
22
23 /// @brief The field is a primary key.
24 ///
25 /// If the field is an auto-incrementable key and not manually set, it is automatically set to the
26 /// next available value on the client side, using a SELECT MAX() query.
27 /// This is happening transparently to the user.
28 ///
29 /// If the field is a GUID, it is automatically set to a new GUID value, if not manually set.
30 ///
31 /// @note If the field is neither auto-incrementable nor a GUID, it must be manually set.
32 AutoAssign,
33
34 /// The field is an integer primary key, and it is auto-incremented by the database.
35 ServerSideAutoIncrement,
36};

245template <detail::FieldElementType T, auto P1, auto P2>
246inline LIGHTWEIGHT_FORCE_INLINE std::string Field<T, P1, P2>::InspectValue() const
247{
248 if constexpr (std::is_same_v<T, std::string>)
249 {
250 std::stringstream result;
251 result << std::quoted(_value, '\'');
252 return result.str();
253 }
254 else if constexpr (std::is_same_v<T, SqlText>)
255 {
256 std::stringstream result;
257 result << std::quoted(_value.value, '\'');
258 return result.str();
259 }
260 else if constexpr (std::is_same_v<T, SqlDate>)
261 return std::format("\'{}\'", _value.value);
262 else if constexpr (std::is_same_v<T, SqlTime>)
263 return std::format("\'{}\'", _value.value);
264 else if constexpr (std::is_same_v<T, SqlDateTime>)
265 return std::format("\'{}\'", _value.value());
266 else if constexpr (SqlNumericType<T>)
267 return std::format("{}", _value.ToString());
268 else if constexpr (requires { _value.has_value(); })
269 {
270 if (_value.has_value())
271 return std::format("{}", _value.value());
272 else
273 return "NULL";
274 }
275 else
276 return std::format("{}", _value);
277}
