197 /// Enables coroutine-based asynchronous methods on this connection.
198 ///
199 /// Wires the connection to an injected execution context: blocking ODBC work is offloaded to
200 /// @p dbWorkers (serialized per connection so the ODBC handle is only ever used by one thread
201 /// at a time) and the awaiting coroutine is resumed via @p resume (typically the application's
202 /// run loop). A native driver-async backend is selected when the driver advertises support;
203 /// otherwise the portable thread-offload backend is used.
204 ///
205 /// Both executors must outlive this connection and every coroutine driven through it.
206 /// Must not be called while asynchronous operations are in flight on this connection (it
207 /// replaces the backend); the connection pool calls @ref DisableAsync on return so each
208 /// re-acquire is a fresh enable rather than a live replacement.
209 ///
210 /// @param dbWorkers The worker-thread pool used to run blocking ODBC calls.
211 /// @param resume The scheduler used to resume coroutines after a blocking step completes.
212 LIGHTWEIGHT_API void EnableAsync(Async::IExecutor& dbWorkers, Async::IResumeScheduler& resume);

214 /// Enables the asynchronous API on this connection using an explicitly provided backend.
215 ///
216 /// This is the dependency-injection seam behind the convenience overload above: callers and
217 /// tests can supply any @ref Async::IAsyncBackend implementation (a fake/inline backend for
218 /// tests, or a future native event backend) instead of the default thread-offload backend.
219 /// The same in-flight / lifetime constraints as the convenience overload apply.
220 ///
221 /// @param backend The async execution backend to install (consumed); must not be null.
222 LIGHTWEIGHT_API void EnableAsync(std::unique_ptr<Async::IAsyncBackend> backend);

224 /// Tears down the asynchronous backend, returning the connection to a non-async state.
225 ///
226 /// Called by the connection pool when a mapper is returned, so a recycled connection never
227 /// carries a stale backend that references executors which may have been destroyed. Safe to
228 /// call when async was never enabled.
229 ///
230 /// @warning Must not be called while async work is in flight on this connection: it destroys the
231 /// backend (and the strand/executors an outstanding offloaded step still references), which would
232 /// race the worker still touching the ODBC handle. Await every async operation on this connection
233 /// to completion before disabling (or before returning the owning pooled @c DataMapper).
234 LIGHTWEIGHT_API void DisableAsync() noexcept;

285inline bool SqlConnection::SupportsNativeRowBatch() const noexcept
286{
287 // Native ODBC parameter-array binding (SQL_ATTR_PARAMSET_SIZE > 1) is a per-driver capability.
288 // Every backend Lightweight supports and tests against honours it; an unverified backend takes the
289 // always-correct per-row path rather than risk a driver that silently ignores the parameter array.
290 switch (ServerType())
291 {
292 case SqlServerType::MICROSOFT_SQL:
293 case SqlServerType::POSTGRESQL:
294 case SqlServerType::SQLITE:
295 return true;
296 case SqlServerType::MYSQL:
297 case SqlServerType::UNKNOWN:
298 return false;
299 }
300 return false;
301}
