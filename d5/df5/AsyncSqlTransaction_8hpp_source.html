18/// A distinct coroutine-based SQL transaction.
19///
20/// Unlike the high- and low-level async methods (which are added directly to the existing
21/// classes), a transaction is a scoped object, so it reads best as its own type. Each
22/// operation is offloaded to the connection's async backend (a worker thread, serialized per
23/// connection) and the awaiting coroutine resumes on the app's resume scheduler.
24///
25/// The underlying connection must have async enabled (SqlConnection::EnableAsync) before use and
26/// must stay async-enabled and alive for the whole lifetime of the transaction: do not destroy the
27/// connection (or return the owning pooled DataMapper, which disables async) between @ref BeginAsync
28/// and the matching @ref CommitAsync / @ref RollbackAsync — the offloaded steps capture the
29/// connection by pointer and would otherwise fail (a clear @c std::logic_error from
30/// @c SqlConnection::AsyncBackend) or dangle.
31///
32/// Always `co_await CommitAsync()` or `co_await RollbackAsync()` explicitly. If the transaction is
33/// still open when destroyed, the destructor performs a best-effort finalization (per the configured
34/// mode) and emits a warning via @c SqlLogger. That finalization is itself routed through the
35/// connection's strand (and blocks the destroying thread until it completes) so it never touches the
36/// ODBC handle concurrently with another in-flight async operation on the same connection.
37///
38/// @warning The destructor's strand-serialized finalization blocks the destroying thread until the
39/// strand has run it, so do not destroy an open transaction from any thread that the strand needs in
40/// order to make progress. That includes (a) destroying it from @e within a strand operation on its own
41/// connection, and (b) — in the multi-threaded model where the resume scheduler @e is the worker pool
42/// that backs the connection's strand — letting it be destroyed while the coroutine is resuming on one
43/// of those worker threads (with a single-worker pool this self-waits and deadlocks). The connection
44/// and its injected executors must also outlive the transaction; tearing the worker pool down first
45/// leaves the finalization undrained and the destructor blocked. Prefer explicit
46/// @ref CommitAsync / @ref RollbackAsync so the destructor never has to finalize.
47///
48/// @code
49/// auto tx = Async::AsyncSqlTransaction { dm.Connection() };
50/// co_await tx.BeginAsync();
51/// co_await dm.UpdateAsync(record);
52/// co_await tx.CommitAsync();
53/// @endcode
54class LIGHTWEIGHT_API AsyncSqlTransaction
55{
56 public:

58 /// @param connection The async-enabled connection to run the transaction on.
59 explicit AsyncSqlTransaction(SqlConnection& connection) noexcept;

67 ~AsyncSqlTransaction();

69 /// Asynchronously begins the transaction (disables auto-commit).
70 ///
71 /// @param defaultMode How an un-finalized transaction is closed on destruction. Defaults to
72 /// @c COMMIT to match the synchronous @ref SqlTransaction, so porting sync code that
73 /// relies on commit-on-scope-exit does not silently switch to rollback.
74 /// @param isolationMode The transaction isolation level.
75 /// @param token Optional cancellation token (checked before the step is dispatched).
76 /// @return A Task that completes once the transaction has begun.
77 /// @throws std::logic_error if a transaction is already open on this object (programmer error).
78 [[nodiscard]] Task<void> BeginAsync(SqlTransactionMode defaultMode = SqlTransactionMode::COMMIT,
79 SqlIsolationMode isolationMode = SqlIsolationMode::DriverDefault,
80 CancellationToken token = {});
81
82 /// Asynchronously commits the transaction.
83 /// @note Finalization is a point of no return and is intentionally @b not cancellable: it always
84 /// runs to completion. (Abandoning it would leave the transaction open, and the destructor
85 /// would then finalize it with the configured default mode.)
86 [[nodiscard]] Task<void> CommitAsync();
87
88 /// Asynchronously rolls back the transaction.
89 /// @note Finalization is a point of no return and is intentionally @b not cancellable: it always
90 /// runs to completion (so a rollback never silently degrades into a commit-on-destruction).
91 [[nodiscard]] Task<void> RollbackAsync();
92
93 private:
94 /// Runs @p finalize (SqlTransaction::Commit or ::Rollback) on the open transaction via the
95 /// connection strand, then clears it. Shared by CommitAsync/RollbackAsync. Not cancellable.
96 /// @param finalize The finalizing member function to invoke.
97 [[nodiscard]] Task<void> FinalizeAsync(void (SqlTransaction::*finalize)());
98
99 SqlConnection* _connection;
100 std::optional<SqlTransaction> _transaction;
101};
