18/// A distinct coroutine-based SQL transaction.
19///
20/// Unlike the high- and low-level async methods (which are added directly to the existing
21/// classes), a transaction is a scoped object, so it reads best as its own type. Each
22/// operation is offloaded to the connection's async backend (a worker thread, serialized per
23/// connection) and the awaiting coroutine resumes on the app's resume scheduler.
24///
25/// The underlying connection must have async enabled (SqlConnection::EnableAsync) before use.
26/// Always `co_await CommitAsync()` or `co_await RollbackAsync()` explicitly; the destructor only
27/// performs a best-effort *synchronous* finalization (per the configured mode) if the transaction
28/// is still open, which runs on the destroying thread and emits a warning via @c SqlLogger.
29///
30/// @warning Because that finalization touches the ODBC connection handle synchronously and *off*
31/// the connection's strand, you must finalize explicitly (`co_await CommitAsync()` /
32/// `RollbackAsync()`) and ensure no other async operation is in flight on the same connection
33/// before this object is destroyed. Destroying an open transaction while a strand operation runs
34/// concurrently on the same connection is a data race on the ODBC handle.
35///
36/// @code
37/// auto tx = Async::AsyncSqlTransaction { dm.Connection() };
38/// co_await tx.BeginAsync();
39/// co_await dm.UpdateAsync(record);
40/// co_await tx.CommitAsync();
41/// @endcode
42class LIGHTWEIGHT_API AsyncSqlTransaction
43{
44 public:
45 /// Constructs an (not-yet-begun) async transaction over @p connection.
46 /// @param connection The async-enabled connection to run the transaction on.
47 explicit AsyncSqlTransaction(SqlConnection& connection) noexcept;
48
49 AsyncSqlTransaction(AsyncSqlTransaction const&) = delete;
50 AsyncSqlTransaction& operator=(AsyncSqlTransaction const&) = delete;
51 AsyncSqlTransaction(AsyncSqlTransaction&&) = delete;
52 AsyncSqlTransaction& operator=(AsyncSqlTransaction&&) = delete;
53
54 /// Best-effort synchronous finalization if still open (see class note).
55 ~AsyncSqlTransaction();
56
57 /// Asynchronously begins the transaction (disables auto-commit).
58 ///
59 /// @param defaultMode How an un-finalized transaction is closed on destruction.
60 /// @param isolationMode The transaction isolation level.
61 /// @return A Task that completes once the transaction has begun.
62 [[nodiscard]] Task<void> BeginAsync(SqlTransactionMode defaultMode = SqlTransactionMode::ROLLBACK,
63 SqlIsolationMode isolationMode = SqlIsolationMode::DriverDefault);
64
65 /// Asynchronously commits the transaction.
66 [[nodiscard]] Task<void> CommitAsync();
67
68 /// Asynchronously rolls back the transaction.
69 [[nodiscard]] Task<void> RollbackAsync();
70
71 private:
72 SqlConnection* _connection;
73 std::optional<SqlTransaction> _transaction;
74};
