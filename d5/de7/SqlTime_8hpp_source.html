41/// Stores the time (of the day) to efficiently write to or read from a database.
42///
43/// @ingroup DataTypes
44struct SqlTime
45{
46 /// The native C++ type for time representation.
47 using native_type = std::chrono::hh_mm_ss<std::chrono::microseconds>;
48
49#if defined(SQL_SS_TIME2)
50 /// The SQL type used for ODBC binding.
51 using sql_type = SQL_SS_TIME2_STRUCT;
52#else
53 /// The SQL type used for ODBC binding.
54 using sql_type = SQL_TIME_STRUCT;
55#endif
56
57 /// The underlying SQL value for ODBC binding.
58 sql_type sqlValue {};
59
60 /// Default constructor.
61 constexpr SqlTime() noexcept = default;
62 /// Move constructor.
63 constexpr SqlTime(SqlTime&&) noexcept = default;
64 /// Move assignment operator.
65 constexpr SqlTime& operator=(SqlTime&&) noexcept = default;
66 /// Copy constructor.
67 constexpr SqlTime(SqlTime const&) noexcept = default;
68 /// Copy assignment operator.
69 constexpr SqlTime& operator=(SqlTime const&) noexcept = default;
70 constexpr ~SqlTime() noexcept = default;
71
72 /// Returns the native time value.
73 [[nodiscard]] LIGHTWEIGHT_FORCE_INLINE constexpr native_type value() const noexcept
74 {
75 return ConvertToNative(sqlValue);
76 }
77
78 /// Equality comparison operator.
79 LIGHTWEIGHT_FORCE_INLINE constexpr bool operator==(SqlTime const& other) const noexcept
80 {
81 return value().to_duration().count() == other.value().to_duration().count();
82 }
83
84 /// Inequality comparison operator.
85 LIGHTWEIGHT_FORCE_INLINE constexpr bool operator!=(SqlTime const& other) const noexcept
86 {
87 return !(*this == other);
88 }
89
90 /// Constructs a SqlTime from a native time value.
91 LIGHTWEIGHT_FORCE_INLINE constexpr SqlTime(native_type value) noexcept:
92 sqlValue { SqlTime::ConvertToSqlValue(value) }
93 {
94 }
95
96 /// Constructs a SqlTime from hours, minutes, seconds, and optional microseconds.
97 LIGHTWEIGHT_FORCE_INLINE constexpr SqlTime(std::chrono::hours hour,
98 std::chrono::minutes minute,
99 std::chrono::seconds second,
100 std::chrono::microseconds micros = {}) noexcept:
101 SqlTime(native_type { hour + minute + second + micros })
102 {
103 }

105 /// Converts a native time value to the SQL representation.
106 static LIGHTWEIGHT_FORCE_INLINE constexpr sql_type ConvertToSqlValue(native_type value) noexcept
107 {
108 return sql_type {
109 .hour = (SQLUSMALLINT) value.hours().count(),
110 .minute = (SQLUSMALLINT) value.minutes().count(),
111 .second = (SQLUSMALLINT) value.seconds().count(),
112#if defined(SQL_SS_TIME2)
113 .fraction = (SQLUINTEGER) value.subseconds().count(),
114#endif
115 };
116 }
117
118 /// Converts a SQL time value to the native representation.
119 static LIGHTWEIGHT_FORCE_INLINE constexpr native_type ConvertToNative(sql_type const& value) noexcept
120 {
121 // clang-format off
122 return native_type { std::chrono::hours { (int) value.hour }
123 + std::chrono::minutes { (unsigned) value.minute }
124 + std::chrono::seconds { (unsigned) value.second }
125#if defined(SQL_SS_TIME2)
126 + std::chrono::microseconds { value.fraction }
127#endif
128
129 };
130 // clang-format on
131 }
132};

134template <>
135struct SqlDataBinder<SqlTime>
136{
137 static constexpr auto ColumnType = SqlColumnTypeDefinitions::Time {};
138
139 // Same legacy-driver fallback as SqlDate (see SqlDate.hpp). The column is
140 // DATETIME on the wire when SQLSRV32 is in use, so bind/fetch as
141 // SQL_C_TIMESTAMP with a synthetic date (1900-01-01 — the ODBC reference
142 // base date) and only round-trip the time-of-day fields. SQL_C_TIME -> DATETIME
143 // is not supported by the legacy driver and would return HYC00.
144 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN InputParameter(SQLHSTMT stmt,
145 SQLUSMALLINT column,
146 SqlTime const& value,
147 SqlDataBinderCallback& cb) noexcept
148 {
149 if (cb.IsLegacyMicrosoftSqlServerDriver())
150 {
151 auto ts = std::make_shared<SQL_TIMESTAMP_STRUCT>();
152 ts->year = 1900;
153 ts->month = 1;
154 ts->day = 1;
155 ts->hour = value.sqlValue.hour;
156 ts->minute = value.sqlValue.minute;
157 ts->second = value.sqlValue.second;
158 ts->fraction = 0; // DATETIME has 1/300s precision; drop sub-second on legacy
159 auto* const rawPtr = ts.get();
160 cb.PlanPostExecuteCallback([ts = std::move(ts)]() {}); // keep buffer alive
161 return SQLBindParameter(stmt,
162 column,
163 SQL_PARAM_INPUT,
164 SQL_C_TIMESTAMP,
165 SQL_TIMESTAMP,
166 23,
167 3,
168 (SQLPOINTER) rawPtr,
169 sizeof(*rawPtr),
170 nullptr);
171 }
172 return SQLBindParameter(
173 stmt, column, SQL_PARAM_INPUT, SQL_C_TYPE_TIME, SQL_TYPE_TIME, 0, 0, (SQLPOINTER) &value.sqlValue, 0, nullptr);
174 }
175
176 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN
177 OutputColumn(SQLHSTMT stmt, SQLUSMALLINT column, SqlTime* result, SQLLEN* indicator, SqlDataBinderCallback& cb) noexcept
178 {
179 if (cb.IsLegacyMicrosoftSqlServerDriver())
180 {
181 auto ts = std::make_shared<SQL_TIMESTAMP_STRUCT>();
182 auto* const rawPtr = ts.get();
183 cb.PlanPostProcessOutputColumn([ts = std::move(ts), result, indicator]() {
184 if (*indicator == SQL_NULL_DATA)
185 {
186 *result = SqlTime {};
187 return;
188 }
189 result->sqlValue.hour = ts->hour;
190 result->sqlValue.minute = ts->minute;
191 result->sqlValue.second = ts->second;
192#if defined(SQL_SS_TIME2)
193 result->sqlValue.fraction = 0;
194#endif
195 });
196 return SQLBindCol(stmt, column, SQL_C_TIMESTAMP, rawPtr, sizeof(*rawPtr), indicator);
197 }
198 return SQLBindCol(stmt, column, SQL_C_TYPE_TIME, &result->sqlValue, sizeof(result->sqlValue), indicator);
199 }
200
201 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN GetColumn(
202 SQLHSTMT stmt, SQLUSMALLINT column, SqlTime* result, SQLLEN* indicator, SqlDataBinderCallback const& cb) noexcept
203 {
204 if (cb.IsLegacyMicrosoftSqlServerDriver())
205 {
206 SQL_TIMESTAMP_STRUCT ts {};
207 auto const rc = SQLGetData(stmt, column, SQL_C_TIMESTAMP, &ts, sizeof(ts), indicator);
208 if (SQL_SUCCEEDED(rc) && *indicator != SQL_NULL_DATA)
209 {
210 result->sqlValue.hour = ts.hour;
211 result->sqlValue.minute = ts.minute;
212 result->sqlValue.second = ts.second;
213#if defined(SQL_SS_TIME2)
214 result->sqlValue.fraction = 0;
215#endif
216 }
217 return rc;
218 }
219 return SQLGetData(stmt, column, SQL_C_TYPE_TIME, &result->sqlValue, sizeof(result->sqlValue), indicator);
220 }
221
222 static LIGHTWEIGHT_FORCE_INLINE std::string Inspect(SqlTime const& value) noexcept
223 {
224 return std::format("{:02}:{:02}:{:02}.{:06}",
225 value.sqlValue.hour,
226 value.sqlValue.minute,
227 value.sqlValue.second,
228 value.sqlValue.fraction);
229 }
