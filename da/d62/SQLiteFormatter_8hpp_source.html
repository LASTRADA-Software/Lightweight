49 [[nodiscard]] std::string StringLiteral(std::string_view value) const noexcept override
50 {
51 if (value.empty())
52 return "''";
53
54 std::string escaped;
55 escaped.reserve(value.size() + 2);
56 escaped += '\'';
57 for (char const c: value)
58 {
59 if (c == '\'')
60 escaped += "''";
61 else
62 escaped += c;
63 }
64 escaped += '\'';
65 return escaped;
66 }

203 [[nodiscard]] virtual std::string BuildColumnDefinition(SqlColumnDeclaration const& column) const
204 {
205 std::stringstream sqlQueryString;
206
207 sqlQueryString << '"' << column.name << "\" ";
208
209 if (column.primaryKey != SqlPrimaryKeyType::AUTO_INCREMENT)
210 sqlQueryString << ColumnType(column.type);
211 else
212 sqlQueryString << ColumnType(SqlColumnTypeDefinitions::Integer {});
213
214 if (column.required)
215 sqlQueryString << " NOT NULL";
216
217 if (column.primaryKey == SqlPrimaryKeyType::AUTO_INCREMENT)
218 sqlQueryString << " PRIMARY KEY AUTOINCREMENT";
219 else if (column.primaryKey == SqlPrimaryKeyType::NONE && !column.index && column.unique)
220 sqlQueryString << " UNIQUE";
221
222 if (!column.defaultValue.empty())
223 sqlQueryString << " DEFAULT " << column.defaultValue;
224
225 return sqlQueryString.str();
226 }

239 [[nodiscard]] StringList CreateTable(std::string_view schema,
240 std::string_view tableName,
241 std::vector<SqlColumnDeclaration> const& columns,
242 std::vector<SqlCompositeForeignKeyConstraint> const& foreignKeys) const override
243 {
244 auto sqlQueries = StringList {};
245
246 sqlQueries.emplace_back([&]() {
247 std::stringstream sqlQueryString;
248 sqlQueryString << "CREATE TABLE \"";
249 if (!schema.empty())
250 {
251 sqlQueryString << schema;
252 sqlQueryString << "\".\"";
253 }
254 sqlQueryString << tableName;
255 sqlQueryString << "\" (";
256 std::vector<SqlColumnDeclaration const*> pks;
257 size_t currentColumn = 0;
258 std::string foreignKeyConstraints;
259 for (SqlColumnDeclaration const& column: columns)
260 {
261 if (currentColumn > 0)
262 sqlQueryString << ",";
263 ++currentColumn;
264 sqlQueryString << "\n ";
265 sqlQueryString << BuildColumnDefinition(column);
266
267 if (column.primaryKey != SqlPrimaryKeyType::NONE)
268 pks.push_back(&column);
269
270 if (column.foreignKey)
271 {
272 foreignKeyConstraints += ",\n ";
273 foreignKeyConstraints += BuildForeignKeyConstraint(column.name, *column.foreignKey);
274 }
275 }
276
277 for (SqlCompositeForeignKeyConstraint const& fk: foreignKeys)
278 {
279 foreignKeyConstraints += ",\n FOREIGN KEY (";
280 for (size_t i = 0; i < fk.columns.size(); ++i)
281 {
282 if (i > 0)
283 foreignKeyConstraints += ", ";
284 foreignKeyConstraints += '"' + fk.columns[i] + '"';
285 }
286 foreignKeyConstraints += ") REFERENCES \"";
287 foreignKeyConstraints += fk.referencedTableName;
288 foreignKeyConstraints += "\" (";
289 for (size_t i = 0; i < fk.referencedColumns.size(); ++i)
290 {
291 if (i > 0)
292 foreignKeyConstraints += ", ";
293 foreignKeyConstraints += '"' + fk.referencedColumns[i] + '"';
294 }
295 foreignKeyConstraints += ")";
296 }
297
298 // Filter out AUTO_INCREMENT from table-level PK constraint if it's the ONLY PK,
299 // because SQLite handles it inline. But for composite keys involving auto-inc (if that's even valid/used),
300 // or if we just want to be explicit.
301 // SQLite restriction: "INTEGER PRIMARY KEY" must be on the column definition for auto-increment.
302 // If we have AUTO_INCREMENT, it's already in BuildColumnDefinition.
303
304 std::erase_if(
305 pks, [](SqlColumnDeclaration const* col) { return col->primaryKey == SqlPrimaryKeyType::AUTO_INCREMENT; });
306
307 if (!pks.empty())
308 {
309 std::ranges::sort(pks, [](SqlColumnDeclaration const* a, SqlColumnDeclaration const* b) {
310 // If both have index, use it. If one has 0 (no index), treat it as "after" indexed ones?
311 // Or just rely on stable sort?
312 // Let's assume indices are properly set for composite keys.
313 // 1-based index vs 0. 0 means "unordered".
314 if (a->primaryKeyIndex != 0 && b->primaryKeyIndex != 0)
315 return a->primaryKeyIndex < b->primaryKeyIndex;
316 if (a->primaryKeyIndex != 0)
317 return true; // a comes first
318 if (b->primaryKeyIndex != 0)
319 return false; // b comes first
320 return false; // keep original order
321 });
322
323 sqlQueryString << ",\n PRIMARY KEY (";
324 for (size_t i = 0; i < pks.size(); ++i)
325 {
326 if (i > 0)
327 sqlQueryString << ", ";
328 sqlQueryString << '"' << pks[i]->name << '"';
329 }
330 sqlQueryString << ")";
331 }
332
333 sqlQueryString << foreignKeyConstraints;
334
335 sqlQueryString << "\n);";
336 return sqlQueryString.str();
337 }());
338
339 for (SqlColumnDeclaration const& column: columns)
340 {
341 if (column.index && column.primaryKey == SqlPrimaryKeyType::NONE)
342 {
343 // primary keys are always indexed
344 if (column.unique)
345 sqlQueries.emplace_back(std::format(R"(CREATE UNIQUE INDEX "{}_{}_index" ON "{}"("{}");)",
346 tableName,
347 column.name,
348 tableName,
349 column.name));
350 else
351 sqlQueries.emplace_back(std::format(
352 R"(CREATE INDEX "{}_{}_index" ON "{}"("{}");)", tableName, column.name, tableName, column.name));
353 }
354 }
355
356 return sqlQueries;
357 }

359 [[nodiscard]] StringList AlterTable(std::string_view schemaName,
360 std::string_view tableName,
361 std::vector<SqlAlterTableCommand> const& commands) const override
362 {
363 std::stringstream sqlQueryString;
364
365 int currentCommand = 0;
366 for (SqlAlterTableCommand const& command: commands)
367 {
368 if (currentCommand > 0)
369 sqlQueryString << '\n';
370 ++currentCommand;
371
372 using namespace SqlAlterTableCommands;
373 sqlQueryString << std::visit(
374 detail::overloaded {
375 [schemaName, tableName](RenameTable const& actualCommand) -> std::string {
376 return std::format(R"(ALTER TABLE {} RENAME TO "{}";)",
377 FormatTableName(schemaName, tableName),
378 actualCommand.newTableName);
379 },
380 [schemaName, tableName, this](AddColumn const& actualCommand) -> std::string {
381 return std::format(R"(ALTER TABLE {} ADD COLUMN "{}" {} {};)",
382 FormatTableName(schemaName, tableName),
383 actualCommand.columnName,
384 ColumnType(actualCommand.columnType),
385 actualCommand.nullable == SqlNullable::NotNull ? "NOT NULL" : "NULL");
386 },
387 [schemaName, tableName, this](AlterColumn const& actualCommand) -> std::string {
388 return std::format(R"(ALTER TABLE {} ALTER COLUMN "{}" {} {};)",
389 FormatTableName(schemaName, tableName),
390 actualCommand.columnName,
391 ColumnType(actualCommand.columnType),
392 actualCommand.nullable == SqlNullable::NotNull ? "NOT NULL" : "NULL");
393 },
394 [schemaName, tableName](RenameColumn const& actualCommand) -> std::string {
395 return std::format(R"(ALTER TABLE {} RENAME COLUMN "{}" TO "{}";)",
396 FormatTableName(schemaName, tableName),
397 actualCommand.oldColumnName,
398 actualCommand.newColumnName);
399 },
400 [schemaName, tableName](DropColumn const& actualCommand) -> std::string {
401 return std::format(R"(ALTER TABLE {} DROP COLUMN "{}";)",
402 FormatTableName(schemaName, tableName),
403 actualCommand.columnName);
404 },
405 [tableName](AddIndex const& actualCommand) -> std::string {
406 using namespace std::string_view_literals;
407 auto const uniqueStr = actualCommand.unique ? "UNIQUE "sv : ""sv;
408 return std::format(R"(CREATE {2}INDEX "{0}_{1}_index" ON "{0}"("{1}");)",
409 tableName,
410 actualCommand.columnName,
411 uniqueStr);
412 },
413 [tableName](DropIndex const& actualCommand) -> std::string {
414 return std::format(R"(DROP INDEX "{0}_{1}_index";)", tableName, actualCommand.columnName);
415 },
416 [schemaName, tableName](AddForeignKey const& actualCommand) -> std::string {
417 return std::format(
418 R"(ALTER TABLE {} ADD {};)",
419 FormatTableName(schemaName, tableName),
420 BuildForeignKeyConstraint(actualCommand.columnName, actualCommand.referencedColumn));
421 },
422 [schemaName, tableName](DropForeignKey const& actualCommand) -> std::string {
423 return std::format(R"(ALTER TABLE {} DROP CONSTRAINT "{}";)",
424 FormatTableName(schemaName, tableName),
425 std::format("FK_{}", actualCommand.columnName));
426 },
427 [tableName](AddCompositeForeignKey const& /*actualCommand*/) -> std::string {
428 // SQLite limitation: ALTER TABLE ADD CONSTRAINT not supported.
429 // We return empty string or comment to satisfy visitor.
430 return std::format(R"(-- AddCompositeForeignKey not supported for {};)", tableName);
431 },
432 },
433 command);
434 }
435
436 return { sqlQueryString.str() };
437 }

439 [[nodiscard]] std::string ColumnType(SqlColumnTypeDefinition const& type) const override
440 {
441 using namespace SqlColumnTypeDefinitions;
442 return std::visit(detail::overloaded {
443 [](Bigint const&) -> std::string { return "BIGINT"; },
444 [](Binary const&) -> std::string { return "BLOB"; },
445 [](Bool const&) -> std::string { return "BOOLEAN"; },
446 [](Char const& type) -> std::string { return std::format("CHAR({})", type.size); },
447 [](Date const&) -> std::string { return "DATE"; },
448 [](DateTime const&) -> std::string { return "DATETIME"; },
449 [](Decimal const& type) -> std::string {
450 return std::format("DECIMAL({}, {})", type.precision, type.scale);
451 },
452 [](Guid const&) -> std::string { return "GUID"; },
453 [](Integer const&) -> std::string { return "INTEGER"; },
454 [](NChar const& type) -> std::string { return std::format("NCHAR({})", type.size); },
455 [](NVarchar const& type) -> std::string { return std::format("NVARCHAR({})", type.size); },
456 [](Real const&) -> std::string { return "REAL"; },
457 [](Smallint const&) -> std::string { return "SMALLINT"; },
458 [](Text const&) -> std::string { return "TEXT"; },
459 [](Time const&) -> std::string { return "TIME"; },
460 [](Timestamp const&) -> std::string { return "TIMESTAMP"; },
461 [](Tinyint const&) -> std::string { return "TINYINT"; },
462 [](VarBinary const& type) -> std::string { return std::format("VARBINARY({})", type.size); },
463 [](Varchar const& type) -> std::string { return std::format("VARCHAR({})", type.size); },
464 },
465 type);
466 }
