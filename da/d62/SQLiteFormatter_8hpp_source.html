160 [[nodiscard]] virtual std::string BuildColumnDefinition(SqlColumnDeclaration const& column) const
161 {
162 std::stringstream sqlQueryString;
163
164 sqlQueryString << '"' << column.name << "\" ";
165
166 if (column.primaryKey != SqlPrimaryKeyType::AUTO_INCREMENT)
167 sqlQueryString << ColumnType(column.type);
168 else
169 sqlQueryString << ColumnType(SqlColumnTypeDefinitions::Integer {});
170
171 if (column.required)
172 sqlQueryString << " NOT NULL";
173
174 if (column.primaryKey == SqlPrimaryKeyType::AUTO_INCREMENT)
175 sqlQueryString << " PRIMARY KEY AUTOINCREMENT";
176 else if (column.primaryKey == SqlPrimaryKeyType::NONE && !column.index && column.unique)
177 sqlQueryString << " UNIQUE";
178
179 return sqlQueryString.str();
180 }

193 [[nodiscard]] StringList CreateTable(std::string_view tableName,
194 std::vector<SqlColumnDeclaration> const& columns) const override
195 {
196 auto sqlQueries = StringList {};
197
198 sqlQueries.emplace_back([&]() {
199 std::stringstream sqlQueryString;
200 sqlQueryString << "CREATE TABLE \"" << tableName << "\" (";
201 size_t currentColumn = 0;
202 std::string primaryKeyColumns;
203 std::string foreignKeyConstraints;
204 for (SqlColumnDeclaration const& column: columns)
205 {
206 if (currentColumn > 0)
207 sqlQueryString << ",";
208 ++currentColumn;
209 sqlQueryString << "\n ";
210 sqlQueryString << BuildColumnDefinition(column);
211 if (column.primaryKey == SqlPrimaryKeyType::MANUAL || column.primaryKey == SqlPrimaryKeyType::GUID)
212 {
213 if (!primaryKeyColumns.empty())
214 primaryKeyColumns += ", ";
215 primaryKeyColumns += '"';
216 primaryKeyColumns += column.name;
217 primaryKeyColumns += '"';
218 }
219 if (column.foreignKey)
220 {
221 foreignKeyConstraints += ",\n ";
222 foreignKeyConstraints += BuildForeignKeyConstraint(column.name, *column.foreignKey);
223 }
224 }
225 if (!primaryKeyColumns.empty())
226 sqlQueryString << ",\n PRIMARY KEY (" << primaryKeyColumns << ")";
227
228 sqlQueryString << foreignKeyConstraints;
229
230 sqlQueryString << "\n);";
231 return sqlQueryString.str();
232 }());
233
234 for (SqlColumnDeclaration const& column: columns)
235 {
236 if (column.index && column.primaryKey == SqlPrimaryKeyType::NONE)
237 {
238 // primary keys are always indexed
239 if (column.unique)
240 sqlQueries.emplace_back(std::format(R"(CREATE UNIQUE INDEX "{}_{}_index" ON "{}"("{}");)",
241 tableName,
242 column.name,
243 tableName,
244 column.name));
245 else
246 sqlQueries.emplace_back(std::format(
247 R"(CREATE INDEX "{}_{}_index" ON "{}"("{}");)", tableName, column.name, tableName, column.name));
248 }
249 }
250
251 return sqlQueries;
252 }

254 [[nodiscard]] StringList AlterTable(std::string_view tableName,
255 std::vector<SqlAlterTableCommand> const& commands) const override
256 {
257 std::stringstream sqlQueryString;
258
259 int currentCommand = 0;
260 for (SqlAlterTableCommand const& command: commands)
261 {
262 if (currentCommand > 0)
263 sqlQueryString << '\n';
264 ++currentCommand;
265
266 using namespace SqlAlterTableCommands;
267 sqlQueryString << std::visit(
268 detail::overloaded {
269 [tableName](RenameTable const& actualCommand) -> std::string {
270 return std::format(R"(ALTER TABLE "{}" RENAME TO "{}";)", tableName, actualCommand.newTableName);
271 },
272 [tableName, this](AddColumn const& actualCommand) -> std::string {
273 return std::format(R"(ALTER TABLE "{}" ADD COLUMN "{}" {} {};)",
274 tableName,
275 actualCommand.columnName,
276 ColumnType(actualCommand.columnType),
277 actualCommand.nullable == SqlNullable::NotNull ? "NOT NULL" : "NULL");
278 },
279 [tableName, this](AlterColumn const& actualCommand) -> std::string {
280 return std::format(R"(ALTER TABLE "{}" ALTER COLUMN "{}" {} {};)",
281 tableName,
282 actualCommand.columnName,
283 ColumnType(actualCommand.columnType),
284 actualCommand.nullable == SqlNullable::NotNull ? "NOT NULL" : "NULL");
285 },
286 [tableName](RenameColumn const& actualCommand) -> std::string {
287 return std::format(R"(ALTER TABLE "{}" RENAME COLUMN "{}" TO "{}";)",
288 tableName,
289 actualCommand.oldColumnName,
290 actualCommand.newColumnName);
291 },
292 [tableName](DropColumn const& actualCommand) -> std::string {
293 return std::format(R"(ALTER TABLE "{}" DROP COLUMN "{}";)", tableName, actualCommand.columnName);
294 },
295 [tableName](AddIndex const& actualCommand) -> std::string {
296 using namespace std::string_view_literals;
297 auto const uniqueStr = actualCommand.unique ? "UNIQUE "sv : ""sv;
298 return std::format(R"(CREATE {2}INDEX "{0}_{1}_index" ON "{0}"("{1}");)",
299 tableName,
300 actualCommand.columnName,
301 uniqueStr);
302 },
303 [tableName](DropIndex const& actualCommand) -> std::string {
304 return std::format(R"(DROP INDEX "{0}_{1}_index";)", tableName, actualCommand.columnName);
305 },
306 [tableName](AddForeignKey const& actualCommand) -> std::string {
307 return std::format(
308 R"(ALTER TABLE "{}" ADD {};)",
309 tableName,
310 BuildForeignKeyConstraint(actualCommand.columnName, actualCommand.referencedColumn));
311 },
312 [tableName](DropForeignKey const& actualCommand) -> std::string {
313 return std::format(R"(ALTER TABLE "{}" DROP CONSTRAINT "{}";)",
314 tableName,
315 std::format("FK_{}", actualCommand.columnName));
316 },
317 },
318 command);
319 }
320
321 return { sqlQueryString.str() };
322 }

324 [[nodiscard]] std::string ColumnType(SqlColumnTypeDefinition const& type) const override
325 {
326 using namespace SqlColumnTypeDefinitions;
327 return std::visit(detail::overloaded {
328 [](Bigint const&) -> std::string { return "BIGINT"; },
329 [](Binary const&) -> std::string { return "BLOB"; },
330 [](Bool const&) -> std::string { return "BOOLEAN"; },
331 [](Char const& type) -> std::string { return std::format("CHAR({})", type.size); },
332 [](Date const&) -> std::string { return "DATE"; },
333 [](DateTime const&) -> std::string { return "DATETIME"; },
334 [](Decimal const& type) -> std::string {
335 return std::format("DECIMAL({}, {})", type.precision, type.scale);
336 },
337 [](Guid const&) -> std::string { return "GUID"; },
338 [](Integer const&) -> std::string { return "INTEGER"; },
339 [](NChar const& type) -> std::string { return std::format("NCHAR({})", type.size); },
340 [](NVarchar const& type) -> std::string { return std::format("NVARCHAR({})", type.size); },
341 [](Real const&) -> std::string { return "REAL"; },
342 [](Smallint const&) -> std::string { return "SMALLINT"; },
343 [](Text const&) -> std::string { return "TEXT"; },
344 [](Time const&) -> std::string { return "TIME"; },
345 [](Timestamp const&) -> std::string { return "TIMESTAMP"; },
346 [](Tinyint const&) -> std::string { return "TINYINT"; },
347 [](VarBinary const& type) -> std::string { return std::format("VARBINARY({})", type.size); },
348 [](Varchar const& type) -> std::string { return std::format("VARCHAR({})", type.size); },
349 },
