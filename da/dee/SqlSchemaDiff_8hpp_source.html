15/// Classifies a schema-level diff entry.
16enum class DiffKind : std::uint8_t
17{
18 OnlyInA, ///< Present only in the left-hand side.

42/// A diff entry for a single table.
43struct TableDiff
44{
45 /// Table name (schema-unqualified). Tables are paired across the two inputs by name only —
46 /// engine-specific schema labels (`dbo`, `public`, `""`) are kept for display but ignored
47 /// for identity, so the same logical table compares as equal across SQL dialects.
48 std::string name;
49
50 /// Schema name on the left-hand side. Empty when @ref kind is @ref DiffKind::OnlyInB
51 /// or when the left-hand engine reports no schema (e.g. SQLite).
52 std::string schemaA;
53
54 /// Schema name on the right-hand side. Empty when @ref kind is @ref DiffKind::OnlyInA
55 /// or when the right-hand engine reports no schema.
56 std::string schemaB;
57
58 /// Table-level classification: @ref DiffKind::OnlyInA / @ref DiffKind::OnlyInB
59 /// when one side lacks the table entirely; @ref DiffKind::Changed when both
60 /// sides have it but column / key / index definitions differ.
61 DiffKind kind {};
62
63 /// Per-column diffs. Populated only when @ref kind is @ref DiffKind::Changed.
64 std::vector<ColumnDiff> columns {};
65
66 /// Human-readable PK differences (e.g. `"primary key columns differ"`).
67 std::vector<std::string> primaryKeyDiffs {};
68
69 /// Human-readable FK differences.
70 std::vector<std::string> foreignKeyDiffs {};
71
72 /// Human-readable index differences.
73 std::vector<std::string> indexDiffs {};
74};

76/// Result of a full schema comparison.
77struct SchemaDiff
78{
79 /// All diff entries, sorted lexicographically by (schema, name).
80 std::vector<TableDiff> tables {};
81
82 /// True if no differences were found.
83 [[nodiscard]] bool Empty() const noexcept
84 {
85 return tables.empty();
86 }

89/// Compares two table lists field-by-field and returns a structural diff.
90///
91/// Pairs tables by **name only** so the same logical table matches across SQL dialects
92/// even when engine-specific schema labels differ (`dbo` vs `public` vs `""`). Both
93/// schema labels are kept on @ref TableDiff for the renderer. Columns are paired by name.
94///
95/// Column comparison uses the canonical @ref Column::type variant (engine-agnostic),
96/// not the dialect-dependent type string, so an `INTEGER` column compares equal whether
97/// the driver reports `int4`, `int`, or `INTEGER`. Other compared fields: nullability,
98/// size, decimal digits, default value, auto-increment, primary-key membership,
99/// foreign-key membership, and uniqueness. Foreign-key constraints are compared at the
100/// table level (not per column).
101///
102/// Pure function — no I/O, no logging, no exceptions.
103LIGHTWEIGHT_API SchemaDiff DiffSchemas(TableList const& a, TableList const& b);
