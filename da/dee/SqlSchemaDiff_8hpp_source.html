15/// Classifies a schema-level diff entry.
16enum class DiffKind : std::uint8_t
17{
18 OnlyInA, ///< Present only in the left-hand side.
19 OnlyInB, ///< Present only in the right-hand side.
20 Changed, ///< Present on both sides but with different definitions.
21};

23/// A diff entry for a single column.
24struct ColumnDiff
25{
26 std::string name;
27 DiffKind kind {};
28
29 /// For @ref DiffKind::Changed: human-readable list of differing fields

33 /// Pointer into the input @ref TableList for the left-hand column. Null when
34 /// @ref kind is @ref DiffKind::OnlyInB.
35 Column const* a = nullptr;
36
37 /// Pointer into the input @ref TableList for the right-hand column. Null when
38 /// @ref kind is @ref DiffKind::OnlyInA.
39 Column const* b = nullptr;
40};

42/// A diff entry for a single table.
43struct TableDiff
44{
45 std::string schema;
46 std::string name;
47
48 /// Table-level classification: @ref DiffKind::OnlyInA / @ref DiffKind::OnlyInB
49 /// when one side lacks the table entirely; @ref DiffKind::Changed when both
50 /// sides have it but column / key / index definitions differ.
51 DiffKind kind {};
52
53 /// Per-column diffs. Populated only when @ref kind is @ref DiffKind::Changed.
54 std::vector<ColumnDiff> columns {};
55
56 /// Human-readable PK differences (e.g. `"primary key columns differ"`).
57 std::vector<std::string> primaryKeyDiffs {};
58
59 /// Human-readable FK differences.
60 std::vector<std::string> foreignKeyDiffs {};
61
62 /// Human-readable index differences.
63 std::vector<std::string> indexDiffs {};
64};
65
66/// Result of a full schema comparison.
67struct SchemaDiff
68{
69 /// All diff entries, sorted lexicographically by (schema, name).
70 std::vector<TableDiff> tables {};
71
72 /// True if no differences were found.
73 [[nodiscard]] bool Empty() const noexcept
74 {
75 return tables.empty();
76 }
77};
