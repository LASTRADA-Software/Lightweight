20 /// Whether @p T's binder exposes an optional-aware row-wise batch binder
21 /// (@c BatchRowWiseInputParameterOptional). True for length-bearing inline inner types
22 /// (fixed-capacity strings), which need a per-row length indicator; false for plain fixed-width inner
23 /// types (primitives, date/time), which carry only a NULL indicator and bind via
24 /// @c BatchInputParameter.
25 template <typename T>
26 concept SqlHasOptionalRowWiseBatchBinder = requires(
27 SQLHSTMT stmt, SQLUSMALLINT column, std::optional<T> const* elem0, std::size_t rowCount, SqlDataBinderCallback& cb) {
28 SqlDataBinder<T>::BatchRowWiseInputParameterOptional(stmt, column, elem0, rowCount, rowCount, cb);
29 };
30} // namespace detail

50 /// Binds an optional column of a native row-wise batch zero-copy, supplying per-row NULL-ness via a
51 /// temporary row-strided indicator buffer.
52 ///
53 /// Dispatches on the inner type @p T:
54 /// - Length-bearing inline types (fixed-capacity strings): delegates to @p T's binder, which fills
55 /// the per-row length-or-@c SQL_NULL_DATA indicator and binds the inline character buffers.
56 /// - Plain fixed-width types (primitives, date/time): the contained value of each row is bound in
57 /// place (no copy); the driver strides by the statement's @c SQL_ATTR_PARAM_BIND_TYPE (== @p
58 /// rowStride). For rows whose optional is disengaged the indicator is @c SQL_NULL_DATA so the
59 /// (indeterminate) value bytes are ignored — the value storage is never dereferenced as a @c T,
60 /// only its address is taken.
61 ///
62 /// @note PRE (guaranteed by the caller): the statement is in row-wise binding mode with
63 /// @c SQL_ATTR_PARAM_BIND_TYPE == @p rowStride, and @c rowStride % alignof(SQLLEN) == 0 so the
64 /// indicator slots at @c i*rowStride stay aligned and non-overlapping.
65 ///
66 /// @param stmt The ODBC statement handle.
67 /// @param column The 1-based parameter column index.
68 /// @param elem0 Address of the @c std::optional<T> within row 0 of the contiguous record array.
69 /// @param rowStride The row-wise bind stride (== sizeof of the row element).
70 /// @param rowCount The number of rows in the batch.
71 /// @param cb The data binder callback (provides the temporary indicator buffer).
72 /// @return The ODBC return code of the underlying bind.
73 static SQLRETURN BatchRowWiseInputParameter(SQLHSTMT stmt,
74 SQLUSMALLINT column,
75 OptionalValue const* elem0,
76 std::size_t rowStride,
77 std::size_t rowCount,
78 SqlDataBinderCallback& cb) noexcept
79 {
80 if constexpr (detail::SqlHasOptionalRowWiseBatchBinder<T>)
81 {
82 // Length-bearing inline inner type (fixed-capacity string): its binder owns the
83 // NULL-or-length indicator fill and the zero-copy bind.
84 return SqlDataBinder<T>::BatchRowWiseInputParameterOptional(stmt, column, elem0, rowStride, rowCount, cb);
85 }
86 else
87 {
88 // Offset of the contained value within std::optional<T> (0 on all known standard libraries,
89 // but computed robustly so the address arithmetic below does not bake in that assumption).
90 OptionalValue const probe { T {} };
91 // Compute the offset through integer addresses rather than pointer subtraction: the two
92 // pointers address distinct objects (the optional and its contained value), so a direct
93 // pointer subtraction would be undefined behavior.
94 auto const valueOffset = reinterpret_cast<std::uintptr_t>(std::addressof(*probe))
95 - reinterpret_cast<std::uintptr_t>(std::addressof(probe));
96
97 // Row-strided indicator buffer: ODBC reads the indicator for row i at base + i*rowStride. The
98 // optionals themselves are embedded in the row structs, so they are also addressed at
99 // sourceBytes + i*rowStride (NOT elem0[i], which would stride by sizeof(std::optional<T>)).
100 auto const* sourceBytes = reinterpret_cast<std::byte const*>(elem0);
101 auto* const indicatorBytes = cb.ProvideBatchStagingBuffer(((rowCount - 1) * rowStride) + sizeof(SQLLEN));
102 for (auto const i: std::views::iota(std::size_t { 0 }, rowCount))
103 {
104 // The optional and its indicator slot for row i both live at offset i*rowStride (row-wise).
105 auto const& optional = *reinterpret_cast<OptionalValue const*>(sourceBytes + (i * rowStride));
106 auto* const indicatorSlot = reinterpret_cast<SQLLEN*>(indicatorBytes + (i * rowStride));
107 *indicatorSlot = optional.has_value() ? SQLLEN { 0 } : SQLLEN { SQL_NULL_DATA };
108 }
109
110 // Zero-copy value bind: point at row 0's contained value; T's batch binder issues the
111 // SQLBindParameter (taking only the address, never forming a T& to possibly-disengaged storage).
112 auto const* valueBase = reinterpret_cast<T const*>(reinterpret_cast<std::byte const*>(elem0) + valueOffset);
113 return SqlDataBinder<T>::BatchInputParameter(
114 stmt, column, valueBase, rowCount, cb, reinterpret_cast<SQLLEN*>(indicatorBytes));
115 }
116 }

118 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN OutputColumn(
119 SQLHSTMT stmt, SQLUSMALLINT column, OptionalValue* result, SQLLEN* indicator, SqlDataBinderCallback& cb) noexcept
120 {
121 if (!result)
122 return SQL_ERROR;
123
124 auto const sqlReturn = SqlDataBinder<T>::OutputColumn(stmt, column, &result->emplace(), indicator, cb);
125 cb.PlanPostProcessOutputColumn([result, indicator]() {
126 if (indicator && *indicator == SQL_NULL_DATA)
127 *result = std::nullopt;
128 });
129 return sqlReturn;
130 }

132 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN GetColumn(SQLHSTMT stmt,
133 SQLUSMALLINT column,
134 OptionalValue* result,
135 SQLLEN* indicator,
136 SqlDataBinderCallback const& cb) noexcept
137 {
138 auto const sqlReturn = SqlDataBinder<T>::GetColumn(stmt, column, &result->emplace(), indicator, cb);
139 if (indicator && *indicator == SQL_NULL_DATA)
140 *result = std::nullopt;
141 return sqlReturn;
142 }

144 static LIGHTWEIGHT_FORCE_INLINE std::string Inspect(OptionalValue const& value) noexcept
145 {
146 if (!value)
147 return "NULL";
148 else
149 return std::string(SqlDataBinder<T>::Inspect(*value));
150 }
151};
