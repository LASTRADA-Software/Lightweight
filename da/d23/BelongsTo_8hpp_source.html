17/// @brief Represents a one-to-one relationship.
18///
19/// The `TheReferencedField` parameter is the field in the other record that references the current record,
20/// in the form of `&OtherRecord::Field`.
21/// Other Field must be a primary key.
22///
23/// @tparam TheReferencedField The field in the other record that references the current record.
24/// @tparam ColumnNameOverrideString If not an empty string, this value will be used as the column name in the database.
25///
26/// @ingroup DataMapper
27///
28/// @code
29/// struct User {
30/// Field<SqlGuid, PrimaryKey::AutoAssign> id;
31/// Field<SqlAnsiString<30>> name;
32/// };
33/// struct Email {
34/// Field<SqlGuid, PrimaryKey::AutoAssign> id;
35/// Field<SqlAnsiString<40>> address;
36/// BelongsTo<&User::id> user;
37/// // Also possible to customize the column name
38/// BelongsTo<&User::id, SqlRealName<"the_user_id">, SqlNullable::Null> maybe_user;
39/// };
40/// @endcode

249 struct Loader
250 {
251 std::function<std::optional<ReferencedRecord>()> loadReference {};
252 };
253
254 /// Used internally to configure on-demand loading of the record.
255 void SetAutoLoader(Loader loader) noexcept
256 {
257 _loader = std::move(loader);
258 }
259
260 private:
261 void RequireLoaded()
262 {
263 if (_loaded)
264 return;
265
266 if (_loader.loadReference)
267 {
268 auto value = _loader.loadReference();
269 if (value)
270 {
271 _record = std::make_unique<ReferencedRecord>(std::move(value.value()));
272 _loaded = true;
273 }
274 }
275
276 if (!_loaded)
277 throw SqlRequireLoadedError(Reflection::TypeNameOf<std::remove_cvref_t<decltype(*this)>>);
278 }
279
280 ValueType _referencedFieldValue {};
281 Loader _loader {};
282 bool _loaded = false;
283 bool _modified = false;
284 std::unique_ptr<ReferencedRecord> _record {};
