3// `ISecretBackend` is the storage-plugin interface behind `SecretResolver`.
4// A backend is typically backed by an OS facility (Windows Credential Manager,
5// macOS Keychain, Linux Secret Service) or a plain-file / env-var / stdin
6// source, and is addressed by a short `Name()` prefix (e.g. `"env"`,
7// `"file"`, `"keychain"`).
8//
9// Backends are expected to be cheap to construct; `SecretResolver` instantiates
10// them once at start-up and holds them for the process lifetime.

23/// Plugin interface implemented by each secret storage backend.
24///
25/// Backends are addressed by the prefix returned from `Name()`. A reference
26/// like `env:MY_PWD` routes to the backend whose `Name()` equals `"env"`.
27/// Bare references (no prefix) walk the backend chain registered with
28/// `SecretResolver::RegisterBackend()`; the first backend that returns a
29/// non-empty result wins.
30class ISecretBackend
31{
32 public:
33 ISecretBackend() = default;
34 ISecretBackend(ISecretBackend const&) = delete;
35 ISecretBackend(ISecretBackend&&) = delete;
36 ISecretBackend& operator=(ISecretBackend const&) = delete;
37 ISecretBackend& operator=(ISecretBackend&&) = delete;
38 virtual ~ISecretBackend() = default;
39
40 /// Short, stable name used both as the prefix (`<name>:`) and as the
41 /// identifier displayed in error messages.
42 [[nodiscard]] virtual std::string_view Name() const noexcept = 0;
43
44 /// Reads the secret identified by `key`. Returns `std::nullopt` when the
45 /// backend does not know `key`. Returns an empty string only if the stored
46 /// value is literally empty.
47 ///
48 /// A backend that is fundamentally unavailable (e.g. the Linux Secret
49 /// Service backend without a running D-Bus session) should also return
50 /// `std::nullopt` so `SecretResolver` can transparently fall through to
51 /// the next backend.
52 [[nodiscard]] virtual std::optional<std::string> Read(std::string_view key) = 0;
53
54 /// Stores `value` under `key`. Not every backend supports writing
55 /// (`env:` is read-only at runtime, for example); those should throw
56 /// `std::runtime_error` with a descriptive message.
57 virtual void Write(std::string_view key, std::string_view value) = 0;
58
59 /// Removes the secret at `key`. Silently succeeds if the key is absent;
60 /// read-only backends should throw just like in `Write`.
61 virtual void Erase(std::string_view key) = 0;
62};
63
64} // namespace Lightweight::Secrets
