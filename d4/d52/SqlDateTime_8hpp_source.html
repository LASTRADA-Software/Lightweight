137 static LIGHTWEIGHT_FORCE_INLINE SQL_TIMESTAMP_STRUCT constexpr ConvertToSqlValue(native_type value) noexcept
138 {
139 using namespace std::chrono;
140 auto const totalDays = floor<days>(value);
141 auto const ymd = year_month_day { totalDays };
142 auto const hms =
143 hh_mm_ss<duration_type> { std::chrono::duration_cast<duration_type>(floor<nanoseconds>(value - totalDays)) };
144 return ConvertToSqlValue(ymd, hms);
145 }
146
147 static LIGHTWEIGHT_FORCE_INLINE SQL_TIMESTAMP_STRUCT constexpr ConvertToSqlValue(
148 std::chrono::year_month_day ymd, std::chrono::hh_mm_ss<duration_type> hms) noexcept
149 {
150 // clang-format off
151 // NB: The fraction field is in 100ns units.
152 return SQL_TIMESTAMP_STRUCT {
153 .year = (SQLSMALLINT) (int) ymd.year(),
154 .month = (SQLUSMALLINT) (unsigned) ymd.month(),
155 .day = (SQLUSMALLINT) (unsigned) ymd.day(),
156 .hour = (SQLUSMALLINT) hms.hours().count(),
157 .minute = (SQLUSMALLINT) hms.minutes().count(),
158 .second = (SQLUSMALLINT) hms.seconds().count(),
159 .fraction = (SQLUINTEGER) (((std::chrono::duration_cast<std::chrono::nanoseconds>(hms.to_duration()).count() % 1'000'000'000llu) / 100) * 100)
160 };
161 // clang-format on
162 }
163
164 static LIGHTWEIGHT_FORCE_INLINE native_type constexpr ConvertToNative(SQL_TIMESTAMP_STRUCT const& time) noexcept
165 {
166 // clang-format off
167 using namespace std::chrono;
168 auto const ymd = year_month_day { std::chrono::year { time.year } / std::chrono::month { time.month } / std::chrono::day { time.day } };
169 auto const hms = hh_mm_ss<duration_type> {
170 duration_cast<duration_type>(
171 hours { time.hour }
172 + minutes { time.minute }
173 + seconds { time.second }
174 + nanoseconds { time.fraction }
175 )
176 };
177 return sys_days { ymd } + hms.to_duration();
178 // clang-format on
179 }

214template <>
215struct std::formatter<SqlDateTime>: std::formatter<std::string>
216{
217 LIGHTWEIGHT_FORCE_INLINE auto format(SqlDateTime const& value, std::format_context& ctx) const
218 -> std::format_context::iterator
219 {
220 // This can be used manually to format the date and time inside sql query builder,
221 // that is why we need to use iso standard 8601, also millisecond precision is used for the formatting
222 //
223 // internally fraction is stored in nanoseconds, here we need to get it in milliseconds
224 auto const milliseconds = value.sqlValue.fraction / 1'000'000;
225 return std::formatter<std::string>::format(std::format("{:04}-{:02}-{:02}T{:02}:{:02}:{:02}.{:03}",
226 value.sqlValue.year,
227 value.sqlValue.month,
228 value.sqlValue.day,
229 value.sqlValue.hour,
230 value.sqlValue.minute,
231 value.sqlValue.second,
232 milliseconds),
233 ctx);
234 }
