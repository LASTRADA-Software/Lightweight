254template <typename Getter, typename Callable>
255constexpr void FixedPointIterate(Getter const& getter, Callable const& callable)
256{
257 auto a = getter();
258 for (;;)
259 {
260 callable(a);
261 auto b = getter();
262 if (a == b)
263 break;
264 a = std::move(b);
265 }

268// Searches upward from current working directory for .test-env.yml
269// Stops at directory containing .git (project root boundary) or filesystem root
270inline std::optional<std::filesystem::path> FindTestEnvFile()
271{
272 auto currentDir = std::filesystem::current_path();
273
274 while (true)
275 {
276 auto testEnvPath = currentDir / ".test-env.yml";
277 if (std::filesystem::exists(testEnvPath))
278 return testEnvPath;
279
280 // Stop at project root (directory containing .git) or filesystem root
281 auto gitPath = currentDir / ".git";
282 if (std::filesystem::exists(gitPath))
283 return std::nullopt; // Reached project root, file not found
284
285 auto parentDir = currentDir.parent_path();
286 if (parentDir == currentDir)
287 return std::nullopt; // Reached filesystem root
288
289 currentDir = parentDir;
290 }
291}

492 static void DropTableRecursively(Lightweight::SqlStatement& stmt,
493 Lightweight::SqlSchema::FullyQualifiedTableName const& table)
494 {
495 auto const dependantTables = Lightweight::SqlSchema::AllForeignKeysTo(stmt, table);
496 for (auto const& dependantTable: dependantTables)
497 DropTableRecursively(stmt, dependantTable.foreignKey.table);
498 stmt.ExecuteDirect(std::format("DROP TABLE IF EXISTS \"{}\"", table.table));
499 }

501 static void DropAllTablesInDatabase(Lightweight::SqlStatement& stmt)
502 {
503 using Lightweight::SqlServerType;
504 switch (stmt.Connection().ServerType())
505 {
506 case SqlServerType::MICROSOFT_SQL:
507 case SqlServerType::MYSQL:
508 stmt.ExecuteDirect(std::format("USE \"{}\"", testDatabaseName));
509 [[fallthrough]];
510 case SqlServerType::SQLITE:
511 case SqlServerType::ORACLE:
512 case SqlServerType::UNKNOWN: {
513 auto const tableNames = GetAllTableNames(stmt);
514 for (auto const& tableName: tableNames)
515 {
516 if (tableName == "sqlite_sequence")
517 continue;
518
519 DropTableRecursively(stmt,
520 Lightweight::SqlSchema::FullyQualifiedTableName {
521 .catalog = {},
522 .schema = {},
523 .table = tableName,
524 });
525 }
526 break;
527 }
528 case SqlServerType::POSTGRESQL:
529 if (m_createdTables.empty())
530 m_createdTables = GetAllTableNames(stmt);
531 for (auto& createdTable: std::views::reverse(m_createdTables))
532 stmt.ExecuteDirect(std::format("DROP TABLE IF EXISTS \"{}\" CASCADE", createdTable));
533 break;
534 }
535 m_createdTables.clear();
536 }

556 {
557 if (stmt.Connection().ServerType() == Lightweight::SqlServerType::ORACLE)
558 return GetAllTableNamesForOracle(stmt);

560 using namespace std::string_literals;
561 auto result = std::vector<std::string>();
562 auto const schemaName = [&] {
563 switch (stmt.Connection().ServerType())

566 return "dbo"s;

575 (SQLSMALLINT) schemaName.size(),

577 0,
578 (SQLCHAR*) "TABLE",
579 SQL_NTS);
580 if (SQL_SUCCEEDED(sqlResult))
581 {
582 while (stmt.FetchRow())
583 {
584 result.emplace_back(stmt.GetColumn<std::string>(3)); // table name
585 }
586 }
587 return result;

673[[nodiscard]] inline std::string NormalizeText(std::string_view const& text)
674{
675 auto result = std::string(text);
676
677 // Remove any newlines and reduce all whitespace to a single space
678 result.erase(std::unique(result.begin(), // NOLINT(modernize-use-ranges)
679 result.end(),
680 [](char a, char b) { return std::isspace(a) && std::isspace(b); }),
681 result.end());
682
683 // trim lading and trailing whitespace
684 while (!result.empty() && std::isspace(result.front()))
685 result.erase(result.begin());
686
687 while (!result.empty() && std::isspace(result.back()))
688 result.pop_back();
689
690 return result;
691}
692
693[[nodiscard]] inline std::string NormalizeText(std::vector<std::string> const& texts)
694{
695 auto result = std::string {};
696 for (auto const& text: texts)
697 {
698 if (!result.empty())
699 result += '\n';
700 result += NormalizeText(text);
701 }
702 return result;
703}
